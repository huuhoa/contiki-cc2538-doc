47 /*
48  * uIP is a small implementation of the IP, UDP and TCP protocols (as
49  * well as some basic ICMP stuff). The implementation couples the IP,
50  * UDP, TCP and the application layers very tightly. To keep the size
51  * of the compiled code down, this code frequently uses the goto
52  * statement. While it would be possible to break the uip_process()
53  * function into many smaller functions, this would increase the code
54  * size because of the overhead of parameter passing and the fact that
55  * the optimier would not be as efficient.
56  *
57  * The principle is that we have a small buffer, called the uip_buf,
58  * in which the device driver puts an incoming packet. The TCP/IP
59  * stack parses the headers in the packet, and calls the
60  * application. If the remote host has sent data to the application,
61  * this data is present in the uip_buf and the application read the
62  * data from there. It is up to the application to put this data into
63  * a byte stream if needed. The application will not be fed with data
64  * that is out of sequence.
65  *
66  * If the application whishes to send data to the peer, it should put
67  * its data into the uip_buf. The uip_appdata pointer points to the
68  * first available byte. The TCP/IP stack will calculate the
69  * checksums, and fill in the necessary header fields and finally send
70  * the packet back to the peer.
71 */

263 static uint16_t
264 chksum(uint16_t sum, const uint8_t *data, uint16_t len)
265 {
266  uint16_t t;
267  const uint8_t *dataptr;
268  const uint8_t *last_byte;
269 
270  dataptr = data;
271  last_byte = data + len - 1;
272 
273  while(dataptr < last_byte) { /* At least two more bytes */
274  t = (dataptr[0] << 8) + dataptr[1];
275  sum += t;
276  if(sum < t) {
277  sum++; /* carry */
278  }
279  dataptr += 2;
280  }
281 
282  if(dataptr == last_byte) {
283  t = (dataptr[0] << 8) + 0;
284  sum += t;
285  if(sum < t) {
286  sum++; /* carry */
287  }
288  }
289 
290  /* Return sum in host byte order. */
291  return sum;
292 }

312 static uint16_t
313 upper_layer_chksum(uint8_t proto)
314 {
315  uint16_t upper_layer_len;
316  uint16_t sum;
317 
318 #if NETSTACK_CONF_WITH_IPV6
319  upper_layer_len = (((uint16_t)(BUF->len[0]) << 8) + BUF->len[1]);
320 #else /* NETSTACK_CONF_WITH_IPV6 */
321  upper_layer_len = (((uint16_t)(BUF->len[0]) << 8) + BUF->len[1]) - UIP_IPH_LEN;
322 #endif /* NETSTACK_CONF_WITH_IPV6 */
323 
324  /* First sum pseudoheader. */
325 
326  /* IP protocol and length fields. This addition cannot carry. */
327  sum = upper_layer_len + proto;
328  /* Sum IP source and destination addresses. */
329  sum = chksum(sum, (uint8_t *)&BUF->srcipaddr, 2 * sizeof(uip_ipaddr_t));
330 
331  /* Sum TCP header and data. */
332  sum = chksum(sum, &uip_buf[UIP_IPH_LEN + UIP_LLH_LEN],
333  upper_layer_len);
334 
335  return (sum == 0) ? 0xffff : uip_htons(sum);
336 }

390 struct uip_conn *
391 uip_connect(const uip_ipaddr_t *ripaddr, uint16_t rport)
392 {
393  register struct uip_conn *conn, *cconn;
394 
395  /* Find an unused local port. */
396  again:
397  ++lastport;
398 
399  if(lastport >= 32000) {
400  lastport = 4096;
401  }
402 
403  /* Check if this port is already in use, and if so try to find
404  another one. */
405  for(c = 0; c < UIP_CONNS; ++c) {
406  conn = &uip_conns[c];
407  if(conn->tcpstateflags != UIP_CLOSED &&
408  conn->lport == uip_htons(lastport)) {
409  goto again;
410  }
411  }
412 
413  conn = 0;
414  for(c = 0; c < UIP_CONNS; ++c) {
415  cconn = &uip_conns[c];
416  if(cconn->tcpstateflags == UIP_CLOSED) {
417  conn = cconn;
418  break;
419  }
420  if(cconn->tcpstateflags == UIP_TIME_WAIT) {
421  if(conn == 0 ||
422  cconn->timer > conn->timer) {
423  conn = cconn;
424  }
425  }
426  }
427 
428  if(conn == 0) {
429  return 0;
430  }
431 
432  conn->tcpstateflags = UIP_SYN_SENT;
433 
434  conn->snd_nxt[0] = iss[0];
435  conn->snd_nxt[1] = iss[1];
436  conn->snd_nxt[2] = iss[2];
437  conn->snd_nxt[3] = iss[3];
438 
439  conn->initialmss = conn->mss = UIP_TCP_MSS;
440 
441  conn->len = 1; /* TCP length of the SYN is one. */
442  conn->nrtx = 0;
443  conn->timer = 1; /* Send the SYN next time around. */
444  conn->rto = UIP_RTO;
445  conn->sa = 0;
446  conn->sv = 16; /* Initial value of the RTT variance. */
447  conn->lport = uip_htons(lastport);
448  conn->rport = rport;
449  uip_ipaddr_copy(&conn->ripaddr, ripaddr);
450 
451  return conn;
452 }

525 #if UIP_REASSEMBLY && !NETSTACK_CONF_WITH_IPV6
526 #define UIP_REASS_BUFSIZE (UIP_BUFSIZE - UIP_LLH_LEN)
527 static uint8_t uip_reassbuf[UIP_REASS_BUFSIZE];
528 static uint8_t uip_reassbitmap[UIP_REASS_BUFSIZE / (8 * 8)];
529 static const uint8_t bitmap_bits[8] = {0xff, 0x7f, 0x3f, 0x1f,
530  0x0f, 0x07, 0x03, 0x01};
531 static uint16_t uip_reasslen;
532 static uint8_t uip_reassflags;
533 #define UIP_REASS_FLAG_LASTFRAG 0x01
534 static uint8_t uip_reasstmr;
535 
536 #define IP_MF 0x20
537 
538 static uint8_t
539 uip_reass(void)
540 {
541  uint16_t offset, len;
542  uint16_t i;
543 
544  /* If ip_reasstmr is zero, no packet is present in the buffer, so we
545  write the IP header of the fragment into the reassembly
546  buffer. The timer is updated with the maximum age. */
547  if(uip_reasstmr == 0) {
548  memcpy(uip_reassbuf, &BUF->vhl, UIP_IPH_LEN);
549  uip_reasstmr = UIP_REASS_MAXAGE;
550  uip_reassflags = 0;
551  /* Clear the bitmap. */
552  memset(uip_reassbitmap, 0, sizeof(uip_reassbitmap));
553  }
554 
555  /* Check if the incoming fragment matches the one currently present
556  in the reasembly buffer. If so, we proceed with copying the
557  fragment into the buffer. */
558  if(BUF->srcipaddr[0] == FBUF->srcipaddr[0] &&
559  BUF->srcipaddr[1] == FBUF->srcipaddr[1] &&
560  BUF->destipaddr[0] == FBUF->destipaddr[0] &&
561  BUF->destipaddr[1] == FBUF->destipaddr[1] &&
562  BUF->ipid[0] == FBUF->ipid[0] &&
563  BUF->ipid[1] == FBUF->ipid[1]) {
564 
565  len = (BUF->len[0] << 8) + BUF->len[1] - (BUF->vhl & 0x0f) * 4;
566  offset = (((BUF->ipoffset[0] & 0x3f) << 8) + BUF->ipoffset[1]) * 8;
567 
568  /* If the offset or the offset + fragment length overflows the
569  reassembly buffer, we discard the entire packet. */
570  if(offset > UIP_REASS_BUFSIZE ||
571  offset + len > UIP_REASS_BUFSIZE) {
572  uip_reasstmr = 0;
573  goto nullreturn;
574  }
575 
576  /* Copy the fragment into the reassembly buffer, at the right
577  offset. */
578  memcpy(&uip_reassbuf[UIP_IPH_LEN + offset],
579  (char *)BUF + (int)((BUF->vhl & 0x0f) * 4),
580  len);
581 
582  /* Update the bitmap. */
583  if(offset / (8 * 8) == (offset + len) / (8 * 8)) {
584  /* If the two endpoints are in the same byte, we only update
585  that byte. */
586 
587  uip_reassbitmap[offset / (8 * 8)] |=
588  bitmap_bits[(offset / 8 ) & 7] &
589  ~bitmap_bits[((offset + len) / 8 ) & 7];
590  } else {
591  /* If the two endpoints are in different bytes, we update the
592  bytes in the endpoints and fill the stuff inbetween with
593  0xff. */
594  uip_reassbitmap[offset / (8 * 8)] |=
595  bitmap_bits[(offset / 8 ) & 7];
596  for(i = 1 + offset / (8 * 8); i < (offset + len) / (8 * 8); ++i) {
597  uip_reassbitmap[i] = 0xff;
598  }
599  uip_reassbitmap[(offset + len) / (8 * 8)] |=
600  ~bitmap_bits[((offset + len) / 8 ) & 7];
601  }
602 
603  /* If this fragment has the More Fragments flag set to zero, we
604  know that this is the last fragment, so we can calculate the
605  size of the entire packet. We also set the
606  IP_REASS_FLAG_LASTFRAG flag to indicate that we have received
607  the final fragment. */
608 
609  if((BUF->ipoffset[0] & IP_MF) == 0) {
610  uip_reassflags |= UIP_REASS_FLAG_LASTFRAG;
611  uip_reasslen = offset + len;
612  }
613 
614  /* Finally, we check if we have a full packet in the buffer. We do
615  this by checking if we have the last fragment and if all bits
616  in the bitmap are set. */
617  if(uip_reassflags & UIP_REASS_FLAG_LASTFRAG) {
618  /* Check all bytes up to and including all but the last byte in
619  the bitmap. */
620  for(i = 0; i < uip_reasslen / (8 * 8) - 1; ++i) {
621  if(uip_reassbitmap[i] != 0xff) {
622  goto nullreturn;
623  }
624  }
625  /* Check the last byte in the bitmap. It should contain just the
626  right amount of bits. */
627  if(uip_reassbitmap[uip_reasslen / (8 * 8)] !=
628  (uint8_t)~bitmap_bits[uip_reasslen / 8 & 7]) {
629  goto nullreturn;
630  }
631 
632  /* If we have come this far, we have a full packet in the
633  buffer, so we allocate a pbuf and copy the packet into it. We
634  also reset the timer. */
635  uip_reasstmr = 0;
636  memcpy(BUF, FBUF, uip_reasslen);
637 
638  /* Pretend to be a "normal" (i.e., not fragmented) IP packet
639  from now on. */
640  BUF->ipoffset[0] = BUF->ipoffset[1] = 0;
641  BUF->len[0] = uip_reasslen >> 8;
642  BUF->len[1] = uip_reasslen & 0xff;
643  BUF->ipchksum = 0;
644  BUF->ipchksum = ~(uip_ipchksum());
645 
646  return uip_reasslen;
647  }
648  }
649 
650  nullreturn:
651  return 0;
652 }
653 #endif /* UIP_REASSEMBLY */
654 /*---------------------------------------------------------------------------*/
655 static void
656 uip_add_rcv_nxt(uint16_t n)
657 {
658  uip_add32(uip_conn->rcv_nxt, n);
659  uip_conn->rcv_nxt[0] = uip_acc32[0];
660  uip_conn->rcv_nxt[1] = uip_acc32[1];
661  uip_conn->rcv_nxt[2] = uip_acc32[2];
662  uip_conn->rcv_nxt[3] = uip_acc32[3];
663 }
664 /*---------------------------------------------------------------------------*/
665 void
666 uip_process(uint8_t flag)
667 {
668  register struct uip_conn *uip_connr = uip_conn;
669 
670 #if UIP_UDP
671  if(flag == UIP_UDP_SEND_CONN) {
672  goto udp_send;
673  }
674 #endif /* UIP_UDP */
675 
676  uip_sappdata = uip_appdata = &uip_buf[UIP_IPTCPH_LEN + UIP_LLH_LEN];
677 
678  /* Check if we were invoked because of a poll request for a
679  particular connection. */
680  if(flag == UIP_POLL_REQUEST) {
681  if((uip_connr->tcpstateflags & UIP_TS_MASK) == UIP_ESTABLISHED &&
682  !uip_outstanding(uip_connr)) {
683  uip_flags = UIP_POLL;
684  UIP_APPCALL();
685  goto appsend;
686 #if UIP_ACTIVE_OPEN && UIP_TCP
687  } else if((uip_connr->tcpstateflags & UIP_TS_MASK) == UIP_SYN_SENT) {
688  /* In the SYN_SENT state, we retransmit out SYN. */
689  BUF->flags = 0;
690  goto tcp_send_syn;
691 #endif /* UIP_ACTIVE_OPEN */
692  }
693  goto drop;
694 
695  /* Check if we were invoked because of the periodic timer firing. */
696  } else if(flag == UIP_TIMER) {
697 #if UIP_REASSEMBLY
698  if(uip_reasstmr != 0) {
699  --uip_reasstmr;
700  }
701 #endif /* UIP_REASSEMBLY */
702  /* Increase the initial sequence number. */
703  if(++iss[3] == 0) {
704  if(++iss[2] == 0) {
705  if(++iss[1] == 0) {
706  ++iss[0];
707  }
708  }
709  }
710 
711  /* Reset the length variables. */
712  uip_len = 0;
713  uip_slen = 0;
714 
715 #if UIP_TCP
716  /* Check if the connection is in a state in which we simply wait
717  for the connection to time out. If so, we increase the
718  connection's timer and remove the connection if it times
719  out. */
720  if(uip_connr->tcpstateflags == UIP_TIME_WAIT ||
721  uip_connr->tcpstateflags == UIP_FIN_WAIT_2) {
722  ++(uip_connr->timer);
723  if(uip_connr->timer == UIP_TIME_WAIT_TIMEOUT) {
724  uip_connr->tcpstateflags = UIP_CLOSED;
725  }
726  } else if(uip_connr->tcpstateflags != UIP_CLOSED) {
727  /* If the connection has outstanding data, we increase the
728  connection's timer and see if it has reached the RTO value
729  in which case we retransmit. */
730 
731  if(uip_outstanding(uip_connr)) {
732  if(uip_connr->timer-- == 0) {
733  if(uip_connr->nrtx == UIP_MAXRTX ||
734  ((uip_connr->tcpstateflags == UIP_SYN_SENT ||
735  uip_connr->tcpstateflags == UIP_SYN_RCVD) &&
736  uip_connr->nrtx == UIP_MAXSYNRTX)) {
737  uip_connr->tcpstateflags = UIP_CLOSED;
738 
739  /* We call UIP_APPCALL() with uip_flags set to
740  UIP_TIMEDOUT to inform the application that the
741  connection has timed out. */
742  uip_flags = UIP_TIMEDOUT;
743  UIP_APPCALL();
744 
745  /* We also send a reset packet to the remote host. */
746  BUF->flags = TCP_RST | TCP_ACK;
747  goto tcp_send_nodata;
748  }
749 
750  /* Exponential backoff. */
751  uip_connr->timer = UIP_RTO << (uip_connr->nrtx > 4?
752  4:
753  uip_connr->nrtx);
754  ++(uip_connr->nrtx);
755 
756  /* Ok, so we need to retransmit. We do this differently
757  depending on which state we are in. In ESTABLISHED, we
758  call upon the application so that it may prepare the
759  data for the retransmit. In SYN_RCVD, we resend the
760  SYNACK that we sent earlier and in LAST_ACK we have to
761  retransmit our FINACK. */
762  UIP_STAT(++uip_stat.tcp.rexmit);
763  switch(uip_connr->tcpstateflags & UIP_TS_MASK) {
764  case UIP_SYN_RCVD:
765  /* In the SYN_RCVD state, we should retransmit our
766  SYNACK. */
767  goto tcp_send_synack;
768 
769 #if UIP_ACTIVE_OPEN
770  case UIP_SYN_SENT:
771  /* In the SYN_SENT state, we retransmit out SYN. */
772  BUF->flags = 0;
773  goto tcp_send_syn;
774 #endif /* UIP_ACTIVE_OPEN */
775 
776  case UIP_ESTABLISHED:
777  /* In the ESTABLISHED state, we call upon the application
778  to do the actual retransmit after which we jump into
779  the code for sending out the packet (the apprexmit
780  label). */
781  uip_flags = UIP_REXMIT;
782  UIP_APPCALL();
783  goto apprexmit;
784 
785  case UIP_FIN_WAIT_1:
786  case UIP_CLOSING:
787  case UIP_LAST_ACK:
788  /* In all these states we should retransmit a FINACK. */
789  goto tcp_send_finack;
790 
791  }
792  }
793  } else if((uip_connr->tcpstateflags & UIP_TS_MASK) == UIP_ESTABLISHED) {
794  /* If there was no need for a retransmission, we poll the
795  application for new data. */
796  uip_flags = UIP_POLL;
797  UIP_APPCALL();
798  goto appsend;
799  }
800  }
801 #endif
802  goto drop;
803  }
804 #if UIP_UDP
805  if(flag == UIP_UDP_TIMER) {
806  if(uip_udp_conn->lport != 0) {
807  uip_conn = NULL;
808  uip_sappdata = uip_appdata = &uip_buf[UIP_LLH_LEN + UIP_IPUDPH_LEN];
809  uip_len = uip_slen = 0;
810  uip_flags = UIP_POLL;
811  UIP_UDP_APPCALL();
812  goto udp_send;
813  } else {
814  goto drop;
815  }
816  }
817 #endif
818 
819  /* This is where the input processing starts. */
820  UIP_STAT(++uip_stat.ip.recv);
821 
822  /* Start of IP input header processing code. */
823 
824 #if NETSTACK_CONF_WITH_IPV6
825  /* Check validity of the IP header. */
826  if((BUF->vtc & 0xf0) != 0x60) { /* IP version and header length. */
827  UIP_STAT(++uip_stat.ip.drop);
828  UIP_STAT(++uip_stat.ip.vhlerr);
829  UIP_LOG("ipv6: invalid version.");
830  goto drop;
831  }
832 #else /* NETSTACK_CONF_WITH_IPV6 */
833  /* Check validity of the IP header. */
834  if(BUF->vhl != 0x45) { /* IP version and header length. */
835  UIP_STAT(++uip_stat.ip.drop);
836  UIP_STAT(++uip_stat.ip.vhlerr);
837  UIP_LOG("ip: invalid version or header length.");
838  goto drop;
839  }
840 #endif /* NETSTACK_CONF_WITH_IPV6 */
841 
842  /* Check the size of the packet. If the size reported to us in
843  uip_len is smaller the size reported in the IP header, we assume
844  that the packet has been corrupted in transit. If the size of
845  uip_len is larger than the size reported in the IP packet header,
846  the packet has been padded and we set uip_len to the correct
847  value. */
848 
849  if((BUF->len[0] << 8) + BUF->len[1] <= uip_len) {
850  uip_len = (BUF->len[0] << 8) + BUF->len[1];
851 #if NETSTACK_CONF_WITH_IPV6
852  uip_len += 40; /* The length reported in the IPv6 header is the
853  length of the payload that follows the
854  header. However, uIP uses the uip_len variable
855  for holding the size of the entire packet,
856  including the IP header. For IPv4 this is not a
857  problem as the length field in the IPv4 header
858  contains the length of the entire packet. But
859  for IPv6 we need to add the size of the IPv6
860  header (40 bytes). */
861 #endif /* NETSTACK_CONF_WITH_IPV6 */
862  } else {
863  UIP_LOG("ip: packet shorter than reported in IP header.");
864  goto drop;
865  }
866 
867 #if !NETSTACK_CONF_WITH_IPV6
868  /* Check the fragment flag. */
869  if((BUF->ipoffset[0] & 0x3f) != 0 ||
870  BUF->ipoffset[1] != 0) {
871 #if UIP_REASSEMBLY
872  uip_len = uip_reass();
873  if(uip_len == 0) {
874  goto drop;
875  }
876 #else /* UIP_REASSEMBLY */
877  UIP_STAT(++uip_stat.ip.drop);
878  UIP_STAT(++uip_stat.ip.fragerr);
879  UIP_LOG("ip: fragment dropped.");
880  goto drop;
881 #endif /* UIP_REASSEMBLY */
882  }
883 #endif /* NETSTACK_CONF_WITH_IPV6 */
884 
885  if(uip_ipaddr_cmp(&uip_hostaddr, &uip_all_zeroes_addr)) {
886  /* If we are configured to use ping IP address configuration and
887  haven't been assigned an IP address yet, we accept all ICMP
888  packets. */
889 #if UIP_PINGADDRCONF && !NETSTACK_CONF_WITH_IPV6
890  if(BUF->proto == UIP_PROTO_ICMP) {
891  UIP_LOG("ip: possible ping config packet received.");
892  goto icmp_input;
893  } else {
894  UIP_LOG("ip: packet dropped since no address assigned.");
895  goto drop;
896  }
897 #endif /* UIP_PINGADDRCONF */
898 
899  } else {
900  /* If IP broadcast support is configured, we check for a broadcast
901  UDP packet, which may be destined to us. */
902 #if UIP_BROADCAST
903  DEBUG_PRINTF("UDP IP checksum 0x%04x\n", uip_ipchksum());
904  if(BUF->proto == UIP_PROTO_UDP &&
905  (uip_ipaddr_cmp(&BUF->destipaddr, &uip_broadcast_addr) ||
906  (BUF->destipaddr.u8[0] & 224) == 224)) { /* XXX this is a
907  hack to be able
908  to receive UDP
909  multicast
910  packets. We check
911  for the bit
912  pattern of the
913  multicast
914  prefix. */
915  goto udp_input;
916  }
917 #endif /* UIP_BROADCAST */
918 
919  /* Check if the packet is destined for our IP address. */
920 #if !NETSTACK_CONF_WITH_IPV6
921  if(!uip_ipaddr_cmp(&BUF->destipaddr, &uip_hostaddr)) {
922  UIP_STAT(++uip_stat.ip.drop);
923  goto drop;
924  }
925 #else /* NETSTACK_CONF_WITH_IPV6 */
926  /* For IPv6, packet reception is a little trickier as we need to
927  make sure that we listen to certain multicast addresses (all
928  hosts multicast address, and the solicited-node multicast
929  address) as well. However, we will cheat here and accept all
930  multicast packets that are sent to the ff02::/16 addresses. */
931  if(!uip_ipaddr_cmp(&BUF->destipaddr, &uip_hostaddr) &&
932  BUF->destipaddr.u16[0] != UIP_HTONS(0xff02)) {
933  UIP_STAT(++uip_stat.ip.drop);
934  goto drop;
935  }
936 #endif /* NETSTACK_CONF_WITH_IPV6 */
937  }
938 
939 #if !NETSTACK_CONF_WITH_IPV6
940  if(uip_ipchksum() != 0xffff) { /* Compute and check the IP header
941  checksum. */
942  UIP_STAT(++uip_stat.ip.drop);
943  UIP_STAT(++uip_stat.ip.chkerr);
944  UIP_LOG("ip: bad checksum.");
945  goto drop;
946  }
947 #endif /* NETSTACK_CONF_WITH_IPV6 */
948 
949 #if UIP_TCP
950  if(BUF->proto == UIP_PROTO_TCP) { /* Check for TCP packet. If so,
951  proceed with TCP input
952  processing. */
953  goto tcp_input;
954  }
955 #endif
956 
957 #if UIP_UDP
958  if(BUF->proto == UIP_PROTO_UDP) {
959  goto udp_input;
960  }
961 #endif /* UIP_UDP */
962 
963 #if !NETSTACK_CONF_WITH_IPV6
964  /* ICMPv4 processing code follows. */
965  if(BUF->proto != UIP_PROTO_ICMP) { /* We only allow ICMP packets from
966  here. */
967  UIP_STAT(++uip_stat.ip.drop);
968  UIP_STAT(++uip_stat.ip.protoerr);
969  UIP_LOG("ip: neither tcp nor icmp.");
970  goto drop;
971  }
972 
973 #if UIP_PINGADDRCONF
974  icmp_input:
975 #endif /* UIP_PINGADDRCONF */
976  UIP_STAT(++uip_stat.icmp.recv);
977 
978  /* ICMP echo (i.e., ping) processing. This is simple, we only change
979  the ICMP type from ECHO to ECHO_REPLY and adjust the ICMP
980  checksum before we return the packet. */
981  if(ICMPBUF->type != ICMP_ECHO) {
982  UIP_STAT(++uip_stat.icmp.drop);
983  UIP_STAT(++uip_stat.icmp.typeerr);
984  UIP_LOG("icmp: not icmp echo.");
985  goto drop;
986  }
987 
988  /* If we are configured to use ping IP address assignment, we use
989  the destination IP address of this ping packet and assign it to
990  ourself. */
991 #if UIP_PINGADDRCONF
992  if(uip_ipaddr_cmp(&uip_hostaddr, &uip_all_zeroes_addr)) {
993  uip_hostaddr = BUF->destipaddr;
994  }
995 #endif /* UIP_PINGADDRCONF */
996 
997  ICMPBUF->type = ICMP_ECHO_REPLY;
998 
999  if(ICMPBUF->icmpchksum >= UIP_HTONS(0xffff - (ICMP_ECHO << 8))) {
1000  ICMPBUF->icmpchksum += UIP_HTONS(ICMP_ECHO << 8) + 1;
1001  } else {
1002  ICMPBUF->icmpchksum += UIP_HTONS(ICMP_ECHO << 8);
1003  }
1004 
1005  /* Swap IP addresses. */
1006  uip_ipaddr_copy(&BUF->destipaddr, &BUF->srcipaddr);
1007  uip_ipaddr_copy(&BUF->srcipaddr, &uip_hostaddr);
1008 
1009  UIP_STAT(++uip_stat.icmp.sent);
1010  BUF->ttl = UIP_TTL;
1011  goto ip_send_nolen;
1012 
1013  /* End of IPv4 input header processing code. */
1014 #else /* !NETSTACK_CONF_WITH_IPV6 */
1015 
1016  /* This is IPv6 ICMPv6 processing code. */
1017  DEBUG_PRINTF("icmp6_input: length %d\n", uip_len);
1018 
1019  if(BUF->proto != UIP_PROTO_ICMP6) { /* We only allow ICMPv6 packets from
1020  here. */
1021  UIP_STAT(++uip_stat.ip.drop);
1022  UIP_STAT(++uip_stat.ip.protoerr);
1023  UIP_LOG("ip: neither tcp nor icmp6.");
1024  goto drop;
1025  }
1026 
1027  UIP_STAT(++uip_stat.icmp.recv);
1028 
1029  /* If we get a neighbor solicitation for our address we should send
1030  a neighbor advertisement message back. */
1031  if(ICMPBUF->type == ICMP6_NEIGHBOR_SOLICITATION) {
1032  if(uip_ipaddr_cmp(&ICMPBUF->icmp6data, &uip_hostaddr)) {
1033 
1034  if(ICMPBUF->options[0] == ICMP6_OPTION_SOURCE_LINK_ADDRESS) {
1035  /* Save the sender's address in our neighbor list. */
1036  uip_neighbor_add(&ICMPBUF->srcipaddr, &(ICMPBUF->options[2]));
1037  }
1038 
1039  /* We should now send a neighbor advertisement back to where the
1040  neighbor solicication came from. */
1041  ICMPBUF->type = ICMP6_NEIGHBOR_ADVERTISEMENT;
1042  ICMPBUF->flags = ICMP6_FLAG_S; /* Solicited flag. */
1043 
1044  ICMPBUF->reserved1 = ICMPBUF->reserved2 = ICMPBUF->reserved3 = 0;
1045 
1046  uip_ipaddr_copy(&ICMPBUF->destipaddr, &ICMPBUF->srcipaddr);
1047  uip_ipaddr_copy(&ICMPBUF->srcipaddr, &uip_hostaddr);
1048  ICMPBUF->options[0] = ICMP6_OPTION_TARGET_LINK_ADDRESS;
1049  ICMPBUF->options[1] = 1; /* Options length, 1 = 8 bytes. */
1050  memcpy(&(ICMPBUF->options[2]), &uip_lladdr, sizeof(uip_lladdr));
1051  ICMPBUF->icmpchksum = 0;
1052  ICMPBUF->icmpchksum = ~uip_icmp6chksum();
1053 
1054  goto send;
1055 
1056  }
1057  goto drop;
1058  } else if(ICMPBUF->type == ICMP6_ECHO) {
1059  /* ICMP echo (i.e., ping) processing. This is simple, we only
1060  change the ICMP type from ECHO to ECHO_REPLY and update the
1061  ICMP checksum before we return the packet. */
1062 
1063  ICMPBUF->type = ICMP6_ECHO_REPLY;
1064 
1065  uip_ipaddr_copy(&BUF->destipaddr, &BUF->srcipaddr);
1066  uip_ipaddr_copy(&BUF->srcipaddr, &uip_hostaddr);
1067  ICMPBUF->icmpchksum = 0;
1068  ICMPBUF->icmpchksum = ~uip_icmp6chksum();
1069 
1070  UIP_STAT(++uip_stat.icmp.sent);
1071  goto send;
1072  } else {
1073  DEBUG_PRINTF("Unknown icmp6 message type %d\n", ICMPBUF->type);
1074  UIP_STAT(++uip_stat.icmp.drop);
1075  UIP_STAT(++uip_stat.icmp.typeerr);
1076  UIP_LOG("icmp: unknown ICMP message.");
1077  goto drop;
1078  }
1079 
1080  /* End of IPv6 ICMP processing. */
1081 
1082 #endif /* !NETSTACK_CONF_WITH_IPV6 */
1083 
1084 #if UIP_UDP
1085  /* UDP input processing. */
1086  udp_input:
1087  /* UDP processing is really just a hack. We don't do anything to the
1088  UDP/IP headers, but let the UDP application do all the hard
1089  work. If the application sets uip_slen, it has a packet to
1090  send. */
1091 #if UIP_UDP_CHECKSUMS
1092  uip_len = uip_len - UIP_IPUDPH_LEN;
1093  uip_appdata = &uip_buf[UIP_LLH_LEN + UIP_IPUDPH_LEN];
1094  if(UDPBUF->udpchksum != 0 && uip_udpchksum() != 0xffff) {
1095  UIP_STAT(++uip_stat.udp.drop);
1096  UIP_STAT(++uip_stat.udp.chkerr);
1097  UIP_LOG("udp: bad checksum.");
1098  goto drop;
1099  }
1100 #else /* UIP_UDP_CHECKSUMS */
1101  uip_len = uip_len - UIP_IPUDPH_LEN;
1102 #endif /* UIP_UDP_CHECKSUMS */
1103 
1104  /* Make sure that the UDP destination port number is not zero. */
1105  if(UDPBUF->destport == 0) {
1106  UIP_LOG("udp: zero port.");
1107  goto drop;
1108  }
1109 
1110  /* Demultiplex this UDP packet between the UDP "connections". */
1111  for(uip_udp_conn = &uip_udp_conns[0];
1112  uip_udp_conn < &uip_udp_conns[UIP_UDP_CONNS];
1113  ++uip_udp_conn) {
1114  /* If the local UDP port is non-zero, the connection is considered
1115  to be used. If so, the local port number is checked against the
1116  destination port number in the received packet. If the two port
1117  numbers match, the remote port number is checked if the
1118  connection is bound to a remote port. Finally, if the
1119  connection is bound to a remote IP address, the source IP
1120  address of the packet is checked. */
1121  if(uip_udp_conn->lport != 0 &&
1122  UDPBUF->destport == uip_udp_conn->lport &&
1123  (uip_udp_conn->rport == 0 ||
1124  UDPBUF->srcport == uip_udp_conn->rport) &&
1125  (uip_ipaddr_cmp(&uip_udp_conn->ripaddr, &uip_all_zeroes_addr) ||
1126  uip_ipaddr_cmp(&uip_udp_conn->ripaddr, &uip_broadcast_addr) ||
1127  uip_ipaddr_cmp(&BUF->srcipaddr, &uip_udp_conn->ripaddr))) {
1128  goto udp_found;
1129  }
1130  }
1131  UIP_LOG("udp: no matching connection found");
1132  UIP_STAT(++uip_stat.udp.drop);
1133 #if UIP_CONF_ICMP_DEST_UNREACH && !NETSTACK_CONF_WITH_IPV6
1134  /* Copy fields from packet header into payload of this ICMP packet. */
1135  memcpy(&(ICMPBUF->payload[0]), ICMPBUF, UIP_IPH_LEN + 8);
1136 
1137  /* Set the ICMP type and code. */
1138  ICMPBUF->type = ICMP_DEST_UNREACHABLE;
1139  ICMPBUF->icode = ICMP_PORT_UNREACHABLE;
1140 
1141  /* Calculate the ICMP checksum. */
1142  ICMPBUF->icmpchksum = 0;
1143  ICMPBUF->icmpchksum = ~uip_chksum((uint16_t *)&(ICMPBUF->type), 36);
1144 
1145  /* Set the IP destination address to be the source address of the
1146  original packet. */
1147  uip_ipaddr_copy(&BUF->destipaddr, &BUF->srcipaddr);
1148 
1149  /* Set our IP address as the source address. */
1150  uip_ipaddr_copy(&BUF->srcipaddr, &uip_hostaddr);
1151 
1152  /* The size of the ICMP destination unreachable packet is 36 + the
1153  size of the IP header (20) = 56. */
1154  uip_len = 36 + UIP_IPH_LEN;
1155  ICMPBUF->len[0] = 0;
1156  ICMPBUF->len[1] = (uint8_t)uip_len;
1157  ICMPBUF->ttl = UIP_TTL;
1158  ICMPBUF->proto = UIP_PROTO_ICMP;
1159 
1160  goto ip_send_nolen;
1161 #else /* UIP_CONF_ICMP_DEST_UNREACH */
1162  goto drop;
1163 #endif /* UIP_CONF_ICMP_DEST_UNREACH */
1164 
1165  udp_found:
1166  UIP_STAT(++uip_stat.udp.recv);
1167  uip_conn = NULL;
1168  uip_flags = UIP_NEWDATA;
1169  uip_sappdata = uip_appdata = &uip_buf[UIP_LLH_LEN + UIP_IPUDPH_LEN];
1170  uip_slen = 0;
1171  UIP_UDP_APPCALL();
1172

1302  ++BUF->ackno[0];

1327  CLOSED connections are found. Thanks to Eddie C. Dost for a very
1328  nice algorithm for the TIME_WAIT search. */

1331  if(uip_conns[c].tcpstateflags == UIP_CLOSED) {

1333  break;

1336  if(uip_connr == 0 ||
1337  uip_conns[c].timer > uip_connr->timer) {
1338  uip_connr = &uip_conns[c];
1339  }
1340  }
1341  }

1345  the remote end will retransmit the packet at a time when we
1346  have more spare connections. */
1347  UIP_STAT(++uip_stat.tcp.syndrop);
1348  UIP_LOG("tcp: found no unused connections.");
1349  goto drop;
1350  }

1352

1356  uip_connr->sv = 4;
1357  uip_connr->nrtx = 0;
1358  uip_connr->lport = BUF->destport;
1359  uip_connr->rport = BUF->srcport;

1366  uip_connr->snd_nxt[3] = iss[3];

1370  uip_connr->rcv_nxt[3] = BUF->seqno[3];

1372  uip_connr->rcv_nxt[1] = BUF->seqno[1];
1373  uip_connr->rcv_nxt[0] = BUF->seqno[0];

1377  if((BUF->tcpoffset & 0xf0) > 0x50) {

1379  opt = uip_buf[UIP_TCPIP_HLEN + UIP_LLH_LEN + c];
1380  if(opt == TCP_OPT_END) {
1381  /* End of options. */
1382  break;
1383  } else if(opt == TCP_OPT_NOOP) {
1384  ++c;
1385  /* NOP option. */
1386  } else if(opt == TCP_OPT_MSS &&
1387  uip_buf[UIP_TCPIP_HLEN + UIP_LLH_LEN + 1 + c] == TCP_OPT_MSS_LEN) {
1388  /* An MSS option with the right option length. */
1389  tmp16 = ((uint16_t)uip_buf[UIP_TCPIP_HLEN + UIP_LLH_LEN + 2 + c] << 8) |
1390  (uint16_t)uip_buf[UIP_IPTCPH_LEN + UIP_LLH_LEN + 3 + c];
1391  uip_connr->initialmss = uip_connr->mss =
1392  tmp16 > UIP_TCP_MSS? UIP_TCP_MSS: tmp16;
1393 
1394  /* And we are done processing options. */
1395  break;
1396  } else {
1397  /* All other options have a length field, so that we easily
1398  can skip past them. */
1399  if(uip_buf[UIP_TCPIP_HLEN + UIP_LLH_LEN + 1 + c] == 0) {
1400  /* If the length field is zero, the options are malformed
1401  and we don't process them further. */
1402  break;
1403  }
1404  c += uip_buf[UIP_TCPIP_HLEN + UIP_LLH_LEN + 1 + c];
1405  }
1406  }
1407  }
1408 
1409  /* Our response will be a SYNACK. */
1410 #if UIP_ACTIVE_OPEN
1411  tcp_send_synack:
1412  BUF->flags = TCP_ACK;
1413 
1414  tcp_send_syn:
1415  BUF->flags |= TCP_SYN;
1416 #else /* UIP_ACTIVE_OPEN */
1417  tcp_send_synack:
1418  BUF->flags = TCP_SYN | TCP_ACK;
1419 #endif /* UIP_ACTIVE_OPEN */
1420

1422  SYNACK. */
1423  BUF->optdata[0] = TCP_OPT_MSS;
1424  BUF->optdata[1] = TCP_OPT_MSS_LEN;
1425  BUF->optdata[2] = (UIP_TCP_MSS) / 256;
1426  BUF->optdata[3] = (UIP_TCP_MSS) & 255;
1427  uip_len = UIP_IPTCPH_LEN + TCP_OPT_MSS_LEN;
1428  BUF->tcpoffset = ((UIP_TCPH_LEN + TCP_OPT_MSS_LEN) / 4) << 4;
1429  goto tcp_send;

1431  /* This label will be jumped to if we found an active connection. */
1432  found:
1433  uip_conn = uip_connr;
1434  uip_flags = 0;
1435  /* We do a very naive form of TCP reset processing; we just accept
1436  any RST and kill our connection. We should in fact check if the
1437  sequence number of this reset is within our advertised window
1438  before we accept the reset. */
1439  if(BUF->flags & TCP_RST) {
1440  uip_connr->tcpstateflags = UIP_CLOSED;
1441  UIP_LOG("tcp: got reset, aborting connection.");
1442  uip_flags = UIP_ABORT;
1443  UIP_APPCALL();
1444  goto drop;
1445  }
1446  /* Calculate the length of the data, if the application has sent
1447  any data to us. */
1448  c = (BUF->tcpoffset >> 4) << 2;
1449  /* uip_len will contain the length of the actual TCP data. This is
1450  calculated by subtracing the length of the TCP header (in
1451  c) and the length of the IP header (20 bytes). */
1452  uip_len = uip_len - c - UIP_IPH_LEN;
1453 
1454  /* First, check if the sequence number of the incoming packet is
1455  what we're expecting next. If not, we send out an ACK with the
1456  correct numbers in, unless we are in the SYN_RCVD state and
1457  receive a SYN, in which case we should retransmit our SYNACK
1458  (which is done futher down). */
1459  if(!((((uip_connr->tcpstateflags & UIP_TS_MASK) == UIP_SYN_SENT) &&
1460  ((BUF->flags & TCP_CTL) == (TCP_SYN | TCP_ACK))) ||
1461  (((uip_connr->tcpstateflags & UIP_TS_MASK) == UIP_SYN_RCVD) &&
1462  ((BUF->flags & TCP_CTL) == TCP_SYN)))) {
1463  if((uip_len > 0 || ((BUF->flags & (TCP_SYN | TCP_FIN)) != 0)) &&
1464  (BUF->seqno[0] != uip_connr->rcv_nxt[0] ||
1465  BUF->seqno[1] != uip_connr->rcv_nxt[1] ||
1466  BUF->seqno[2] != uip_connr->rcv_nxt[2] ||
1467  BUF->seqno[3] != uip_connr->rcv_nxt[3])) {
1468  goto tcp_send_ack;
1469  }
1470  }
1471 
1472  /* Next, check if the incoming segment acknowledges any outstanding
1473  data. If so, we update the sequence number, reset the length of
1474  the outstanding data, calculate RTT estimations, and reset the
1475  retransmission timer. */
1476  if((BUF->flags & TCP_ACK) && uip_outstanding(uip_connr)) {
1477  uip_add32(uip_connr->snd_nxt, uip_connr->len);
1478 
1479  if(BUF->ackno[0] == uip_acc32[0] &&
1480  BUF->ackno[1] == uip_acc32[1] &&
1481  BUF->ackno[2] == uip_acc32[2] &&
1482  BUF->ackno[3] == uip_acc32[3]) {
1483  /* Update sequence number. */
1484  uip_connr->snd_nxt[0] = uip_acc32[0];
1485  uip_connr->snd_nxt[1] = uip_acc32[1];
1486  uip_connr->snd_nxt[2] = uip_acc32[2];
1487  uip_connr->snd_nxt[3] = uip_acc32[3];
1488 
1489  /* Do RTT estimation, unless we have done retransmissions. */
1490  if(uip_connr->nrtx == 0) {
1491  signed char m;
1492  m = uip_connr->rto - uip_connr->timer;
1493  /* This is taken directly from VJs original code in his paper */
1494  m = m - (uip_connr->sa >> 3);
1495  uip_connr->sa += m;
1496  if(m < 0) {
1497  m = -m;
1498  }
1499  m = m - (uip_connr->sv >> 2);
1500  uip_connr->sv += m;
1501  uip_connr->rto = (uip_connr->sa >> 3) + uip_connr->sv;
1502 
1503  }
1504  /* Set the acknowledged flag. */
1505  uip_flags = UIP_ACKDATA;
1506  /* Reset the retransmission timer. */
1507  uip_connr->timer = uip_connr->rto;
1508 
1509  /* Reset length of outstanding data. */
1510  uip_connr->len = 0;
1511  }
1512 
1513  }
1514 
1515  /* Do different things depending on in what state the connection is. */
1516  switch(uip_connr->tcpstateflags & UIP_TS_MASK) {
1517  /* CLOSED and LISTEN are not handled here. CLOSE_WAIT is not
1518  implemented, since we force the application to close when the
1519  peer sends a FIN (hence the application goes directly from
1520  ESTABLISHED to LAST_ACK). */
1521  case UIP_SYN_RCVD:
1522  /* In SYN_RCVD we have sent out a SYNACK in response to a SYN, and
1523  we are waiting for an ACK that acknowledges the data we sent
1524  out the last time. Therefore, we want to have the UIP_ACKDATA
1525  flag set. If so, we enter the ESTABLISHED state. */
1526  if(uip_flags & UIP_ACKDATA) {
1527  uip_connr->tcpstateflags = UIP_ESTABLISHED;
1528  uip_flags = UIP_CONNECTED;
1529  uip_connr->len = 0;
1530  if(uip_len > 0) {
1531  uip_flags |= UIP_NEWDATA;
1532  uip_add_rcv_nxt(uip_len);
1533  }
1534  uip_slen = 0;
1535  UIP_APPCALL();
1536  goto appsend;
1537  }
1538  /* We need to retransmit the SYNACK */
1539  if((BUF->flags & TCP_CTL) == TCP_SYN) {
1540  goto tcp_send_synack;
1541  }
1542  goto drop;
1543 #if UIP_ACTIVE_OPEN
1544  case UIP_SYN_SENT:
1545  /* In SYN_SENT, we wait for a SYNACK that is sent in response to
1546  our SYN. The rcv_nxt is set to sequence number in the SYNACK
1547  plus one, and we send an ACK. We move into the ESTABLISHED
1548  state. */
1549  if((uip_flags & UIP_ACKDATA) &&
1550  (BUF->flags & TCP_CTL) == (TCP_SYN | TCP_ACK)) {
1551

1553  if((BUF->tcpoffset & 0xf0) > 0x50) {

1555  opt = uip_buf[UIP_IPTCPH_LEN + UIP_LLH_LEN + c];
1556  if(opt == TCP_OPT_END) {
1557  /* End of options. */
1558  break;
1559  } else if(opt == TCP_OPT_NOOP) {

1561  /* NOP option. */
1562  } else if(opt == TCP_OPT_MSS &&
1563  uip_buf[UIP_TCPIP_HLEN + UIP_LLH_LEN + 1 + c] == TCP_OPT_MSS_LEN) {
1564  /* An MSS option with the right option length. */
1565  tmp16 = (uip_buf[UIP_TCPIP_HLEN + UIP_LLH_LEN + 2 + c] << 8) |
1566  uip_buf[UIP_TCPIP_HLEN + UIP_LLH_LEN + 3 + c];

1568  uip_connr->mss = tmp16 > UIP_TCP_MSS? UIP_TCP_MSS: tmp16;
1569

1571  break;

1574  can skip past them. */
1575  if(uip_buf[UIP_TCPIP_HLEN + UIP_LLH_LEN + 1 + c] == 0) {
1576  /* If the length field is zero, the options are malformed
1577  and we don't process them further. */
1578  break;
1579  }
1580  c += uip_buf[UIP_TCPIP_HLEN + UIP_LLH_LEN + 1 + c];
1581  }
1582  }
1583  }

1585  uip_connr->rcv_nxt[0] = BUF->seqno[0];
1586  uip_connr->rcv_nxt[1] = BUF->seqno[1];

1588  uip_connr->rcv_nxt[3] = BUF->seqno[3];

1596  }

1598  uip_flags = UIP_ABORT;
1599  UIP_APPCALL();
1600  /* The connection is closed after we send the RST */
1601  uip_conn->tcpstateflags = UIP_CLOSED;
1602  goto reset;
1603 #endif /* UIP_ACTIVE_OPEN */
1604 
1605  case UIP_ESTABLISHED:
1606  /* In the ESTABLISHED state, we call upon the application to feed
1607  data into the uip_buf. If the UIP_ACKDATA flag is set, the
1608  application should put new data into the buffer, otherwise we are
1609  retransmitting an old segment, and the application should put that
1610  data into the buffer.
1611 
1612  If the incoming packet is a FIN, we should close the connection on
1613  this side as well, and we send out a FIN and enter the LAST_ACK
1614  state. We require that there is no outstanding data; otherwise the
1615  sequence numbers will be screwed up. */
1616 
1617  if(BUF->flags & TCP_FIN && !(uip_connr->tcpstateflags & UIP_STOPPED)) {
1618  if(uip_outstanding(uip_connr)) {
1619  goto drop;
1620  }
1621  uip_add_rcv_nxt(1 + uip_len);
1622  uip_flags |= UIP_CLOSE;
1623  if(uip_len > 0) {
1624  uip_flags |= UIP_NEWDATA;
1625  }
1626  UIP_APPCALL();
1627  uip_connr->len = 1;
1628  uip_connr->tcpstateflags = UIP_LAST_ACK;
1629  uip_connr->nrtx = 0;
1630  tcp_send_finack:
1631  BUF->flags = TCP_FIN | TCP_ACK;
1632  goto tcp_send_nodata;
1633  }
1634 
1635  /* Check the URG flag. If this is set, the segment carries urgent
1636  data that we must pass to the application. */
1637  if((BUF->flags & TCP_URG) != 0) {
1638 #if UIP_URGDATA > 0
1639  uip_urglen = (BUF->urgp[0] << 8) | BUF->urgp[1];
1640  if(uip_urglen > uip_len) {
1641  /* There is more urgent data in the next segment to come. */
1642  uip_urglen = uip_len;
1643  }
1644  uip_add_rcv_nxt(uip_urglen);
1645  uip_len -= uip_urglen;
1646  uip_urgdata = uip_appdata;
1647  uip_appdata += uip_urglen;
1648  } else {
1649  uip_urglen = 0;
1650 #else /* UIP_URGDATA > 0 */
1651  uip_appdata = ((char *)uip_appdata) + ((BUF->urgp[0] << 8) | BUF->urgp[1]);
1652  uip_len -= (BUF->urgp[0] << 8) | BUF->urgp[1];
1653 #endif /* UIP_URGDATA > 0 */
1654  }
1655 
1656  /* If uip_len > 0 we have TCP data in the packet, and we flag this
1657  by setting the UIP_NEWDATA flag and update the sequence number
1658  we acknowledge. If the application has stopped the dataflow
1659  using uip_stop(), we must not accept any data packets from the
1660  remote host. */
1661  if(uip_len > 0 && !(uip_connr->tcpstateflags & UIP_STOPPED)) {
1662  uip_flags |= UIP_NEWDATA;
1663  uip_add_rcv_nxt(uip_len);
1664  }
1665 
1666  /* Check if the available buffer space advertised by the other end
1667  is smaller than the initial MSS for this connection. If so, we
1668  set the current MSS to the window size to ensure that the
1669  application does not send more data than the other end can
1670  handle.
1671 
1672  If the remote host advertises a zero window, we set the MSS to
1673  the initial MSS so that the application will send an entire MSS
1674  of data. This data will not be acknowledged by the receiver,
1675  and the application will retransmit it. This is called the
1676  "persistent timer" and uses the retransmission mechanim.
1677  */
1678  tmp16 = ((uint16_t)BUF->wnd[0] << 8) + (uint16_t)BUF->wnd[1];
1679  if(tmp16 > uip_connr->initialmss ||
1680  tmp16 == 0) {
1681  tmp16 = uip_connr->initialmss;
1682  }
1683  uip_connr->mss = tmp16;
1684 
1685  /* If this packet constitutes an ACK for outstanding data (flagged
1686  by the UIP_ACKDATA flag, we should call the application since it
1687  might want to send more data. If the incoming packet had data
1688  from the peer (as flagged by the UIP_NEWDATA flag), the
1689  application must also be notified.
1690 
1691  When the application is called, the global variable uip_len
1692  contains the length of the incoming data. The application can
1693  access the incoming data through the global pointer
1694  uip_appdata, which usually points UIP_IPTCPH_LEN + UIP_LLH_LEN
1695  bytes into the uip_buf array.
1696 
1697  If the application wishes to send any data, this data should be
1698  put into the uip_appdata and the length of the data should be
1699  put into uip_len. If the application don't have any data to
1700  send, uip_len must be set to 0. */
1701  if(uip_flags & (UIP_NEWDATA | UIP_ACKDATA)) {
1702  uip_slen = 0;
1703  UIP_APPCALL();
1704 
1705  appsend:
1706 
1707  if(uip_flags & UIP_ABORT) {
1708  uip_slen = 0;
1709  uip_connr->tcpstateflags = UIP_CLOSED;
1710  BUF->flags = TCP_RST | TCP_ACK;
1711  goto tcp_send_nodata;
1712  }
1713 
1714  if(uip_flags & UIP_CLOSE) {
1715  uip_slen = 0;
1716  uip_connr->len = 1;
1717  uip_connr->tcpstateflags = UIP_FIN_WAIT_1;
1718  uip_connr->nrtx = 0;
1719  BUF->flags = TCP_FIN | TCP_ACK;
1720  goto tcp_send_nodata;
1721  }
1722 
1723  /* If uip_slen > 0, the application has data to be sent. */
1724  if(uip_slen > 0) {
1725 
1726  /* If the connection has acknowledged data, the contents of
1727  the ->len variable should be discarded. */
1728  if((uip_flags & UIP_ACKDATA) != 0) {
1729  uip_connr->len = 0;
1730  }
1731 
1732  /* If the ->len variable is non-zero the connection has
1733  already data in transit and cannot send anymore right
1734  now. */
1735  if(uip_connr->len == 0) {
1736 
1737  /* The application cannot send more than what is allowed by
1738  the mss (the minumum of the MSS and the available
1739  window). */
1740  if(uip_slen > uip_connr->mss) {
1741  uip_slen = uip_connr->mss;
1742  }
1743 
1744  /* Remember how much data we send out now so that we know
1745  when everything has been acknowledged. */
1746  uip_connr->len = uip_slen;
1747  } else {
1748 
1749  /* If the application already had unacknowledged data, we
1750  make sure that the application does not send (i.e.,
1751  retransmit) out more than it previously sent out. */
1752  uip_slen = uip_connr->len;
1753  }
1754  }
1755  uip_connr->nrtx = 0;
1756  apprexmit:
1757  uip_appdata = uip_sappdata;
1758 
1759  /* If the application has data to be sent, or if the incoming
1760  packet had new data in it, we must send out a packet. */
1761  if(uip_slen > 0 && uip_connr->len > 0) {
1762  /* Add the length of the IP and TCP headers. */
1763  uip_len = uip_connr->len + UIP_TCPIP_HLEN;
1764  /* We always set the ACK flag in response packets. */
1765  BUF->flags = TCP_ACK | TCP_PSH;
1766  /* Send the packet. */
1767  goto tcp_send_noopts;
1768  }
1769  /* If there is no data to send, just send out a pure ACK if
1770  there is newdata. */
1771  if(uip_flags & UIP_NEWDATA) {
1772  uip_len = UIP_TCPIP_HLEN;
1773  BUF->flags = TCP_ACK;
1774  goto tcp_send_noopts;
1775  }
1776  }
1777  goto drop;
1778  case UIP_LAST_ACK:
1779  /* We can close this connection if the peer has acknowledged our
1780  FIN. This is indicated by the UIP_ACKDATA flag. */
1781  if(uip_flags & UIP_ACKDATA) {
1782  uip_connr->tcpstateflags = UIP_CLOSED;
1783  uip_flags = UIP_CLOSE;
1784  UIP_APPCALL();
1785  }
1786  break;
1787 
1788  case UIP_FIN_WAIT_1:
1789  /* The application has closed the connection, but the remote host
1790  hasn't closed its end yet. Thus we do nothing but wait for a
1791  FIN from the other side. */
1792  if(uip_len > 0) {
1793  uip_add_rcv_nxt(uip_len);
1794  }
1795  if(BUF->flags & TCP_FIN) {
1796  if(uip_flags & UIP_ACKDATA) {
1797  uip_connr->tcpstateflags = UIP_TIME_WAIT;
1798  uip_connr->timer = 0;
1799  uip_connr->len = 0;
1800  } else {
1801  uip_connr->tcpstateflags = UIP_CLOSING;
1802  }
1803  uip_add_rcv_nxt(1);
1804  uip_flags = UIP_CLOSE;
1805  UIP_APPCALL();
1806  goto tcp_send_ack;
1807  } else if(uip_flags & UIP_ACKDATA) {
1808  uip_connr->tcpstateflags = UIP_FIN_WAIT_2;
1809  uip_connr->len = 0;
1810  goto drop;
1811  }
1812  if(uip_len > 0) {
1813  goto tcp_send_ack;
1814  }
1815  goto drop;
1816 
1817  case UIP_FIN_WAIT_2:
1818  if(uip_len > 0) {
1819  uip_add_rcv_nxt(uip_len);
1820  }
1821  if(BUF->flags & TCP_FIN) {
1822  uip_connr->tcpstateflags = UIP_TIME_WAIT;
1823  uip_connr->timer = 0;
1824  uip_add_rcv_nxt(1);
1825  uip_flags = UIP_CLOSE;
1826  UIP_APPCALL();
1827  goto tcp_send_ack;
1828  }
1829  if(uip_len > 0) {
1830  goto tcp_send_ack;
1831  }
1832  goto drop;
1833 
1834  case UIP_TIME_WAIT:
1835  goto tcp_send_ack;
1836 
1837  case UIP_CLOSING:
1838  if(uip_flags & UIP_ACKDATA) {
1839  uip_connr->tcpstateflags = UIP_TIME_WAIT;
1840  uip_connr->timer = 0;
1841  }
1842  }
1843  goto drop;
1844 
1845  /* We jump here when we are ready to send the packet, and just want
1846  to set the appropriate TCP sequence numbers in the TCP header. */
1847  tcp_send_ack:
1848  BUF->flags = TCP_ACK;
1849 
1850  tcp_send_nodata:
1851  uip_len = UIP_IPTCPH_LEN;
1852 
1853  tcp_send_noopts:
1854  BUF->tcpoffset = (UIP_TCPH_LEN / 4) << 4;
1855 
1856  /* We're done with the input processing. We are now ready to send a
1857  reply. Our job is to fill in all the fields of the TCP and IP
1858  headers before calculating the checksum and finally send the
1859  packet. */
1860  tcp_send:
1861  BUF->ackno[0] = uip_connr->rcv_nxt[0];
1862  BUF->ackno[1] = uip_connr->rcv_nxt[1];
1863  BUF->ackno[2] = uip_connr->rcv_nxt[2];
1864  BUF->ackno[3] = uip_connr->rcv_nxt[3];
1865 
1866  BUF->seqno[0] = uip_connr->snd_nxt[0];
1867  BUF->seqno[1] = uip_connr->snd_nxt[1];
1868  BUF->seqno[2] = uip_connr->snd_nxt[2];
1869  BUF->seqno[3] = uip_connr->snd_nxt[3];
1870 
1871  BUF->srcport = uip_connr->lport;
1872  BUF->destport = uip_connr->rport;
1873 
1874  uip_ipaddr_copy(&BUF->srcipaddr, &uip_hostaddr);
1875  uip_ipaddr_copy(&BUF->destipaddr, &uip_connr->ripaddr);
1876 
1877  if(uip_connr->tcpstateflags & UIP_STOPPED) {
1878  /* If the connection has issued uip_stop(), we advertise a zero
1879  window so that the remote host will stop sending data. */
1880  BUF->wnd[0] = BUF->wnd[1] = 0;
1881  } else {
1882  BUF->wnd[0] = ((UIP_RECEIVE_WINDOW) >> 8);
1883  BUF->wnd[1] = ((UIP_RECEIVE_WINDOW) & 0xff);
1884  }
1885 
1886  tcp_send_noconn:
1887  BUF->proto = UIP_PROTO_TCP;
1888 
1889  BUF->ttl = UIP_TTL;
1890 #if NETSTACK_CONF_WITH_IPV6
1891  /* For IPv6, the IP length field does not include the IPv6 IP header
1892  length. */
1893  BUF->len[0] = ((uip_len - UIP_IPH_LEN) >> 8);
1894  BUF->len[1] = ((uip_len - UIP_IPH_LEN) & 0xff);
1895 #else /* NETSTACK_CONF_WITH_IPV6 */
1896  BUF->len[0] = (uip_len >> 8);
1897  BUF->len[1] = (uip_len & 0xff);
1898 #endif /* NETSTACK_CONF_WITH_IPV6 */
1899 
1900  BUF->urgp[0] = BUF->urgp[1] = 0;
1901 
1902  /* Calculate TCP checksum. */
1903  BUF->tcpchksum = 0;
1904  BUF->tcpchksum = ~(uip_tcpchksum());
1905 #endif
1906 
1907  ip_send_nolen:
1908 #if NETSTACK_CONF_WITH_IPV6
1909  BUF->vtc = 0x60;
1910  BUF->tcflow = 0x00;
1911  BUF->flow = 0x00;
1912 #else /* NETSTACK_CONF_WITH_IPV6 */
1913  BUF->vhl = 0x45;
1914  BUF->tos = 0;
1915  BUF->ipoffset[0] = BUF->ipoffset[1] = 0;
1916  ++ipid;
1917  BUF->ipid[0] = ipid >> 8;
1918  BUF->ipid[1] = ipid & 0xff;
1919  /* Calculate IP checksum. */
1920  BUF->ipchksum = 0;
1921  BUF->ipchksum = ~(uip_ipchksum());
1922  DEBUG_PRINTF("uip ip_send_nolen: chkecum 0x%04x\n", uip_ipchksum());
1923 #endif /* NETSTACK_CONF_WITH_IPV6 */
1924  UIP_STAT(++uip_stat.tcp.sent);
1925 #if NETSTACK_CONF_WITH_IPV6
1926  send:
1927 #endif /* NETSTACK_CONF_WITH_IPV6 */
1928  DEBUG_PRINTF("Sending packet with length %d (%d)\n", uip_len,
1929  (BUF->len[0] << 8) | BUF->len[1]);
1930 
1931  UIP_STAT(++uip_stat.ip.sent);
1932  /* Return and let the caller do the actual transmission. */
1933  uip_flags = 0;
1934  return;
1935 
1936  drop:
1937  uip_len = 0;
1938  uip_flags = 0;
1939  return;
1940 }
