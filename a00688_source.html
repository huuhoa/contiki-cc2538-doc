76 static uip_ipaddr_t *
77 set_global_address(void)
78 {
79  static uip_ipaddr_t ipaddr;
80  int i;
81  uint8_t state;
82 
83  uip_ip6addr(&ipaddr, 0xaaaa, 0, 0, 0, 0, 0, 0, 0);
84  uip_ds6_set_addr_iid(&ipaddr, &uip_lladdr);
85  uip_ds6_addr_add(&ipaddr, 0, ADDR_AUTOCONF);
86 
87  printf("IPv6 addresses: ");
88  for(i = 0; i < UIP_DS6_ADDR_NB; i++) {
89  state = uip_ds6_if.addr_list[i].state;
90  if(uip_ds6_if.addr_list[i].isused &&
91  (state == ADDR_TENTATIVE || state == ADDR_PREFERRED)) {
92  uip_debug_ipaddr_print(&uip_ds6_if.addr_list[i].ipaddr);
93  printf("\n");
94  }
95  }
96 
97  return &ipaddr;
98 }

100 static void
101 create_rpl_dag(uip_ipaddr_t *ipaddr)
102 {
103  struct uip_ds6_addr *root_if;
104 
105  root_if = uip_ds6_addr_lookup(ipaddr);
106  if(root_if != NULL) {
107  rpl_dag_t *dag;
108  uip_ipaddr_t prefix;
109 
110  rpl_set_root(RPL_DEFAULT_INSTANCE, ipaddr);
111  dag = rpl_get_any_dag();
112  uip_ip6addr(&prefix, 0xaaaa, 0, 0, 0, 0, 0, 0, 0);
113  rpl_set_prefix(dag, &prefix, 64);
114  PRINTF("created a new RPL dag\n");
115  } else {
116  PRINTF("failed to create a new RPL DAG\n");
117  }
118 }
