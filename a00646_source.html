33  * \file
34  * A simple demo project which demonstrates the cc2531 USB dongle
35  * USB (CDC_ACM) functionality.

37  * It will print out periodically. Anything you type in the dongle's

44 #include "contiki.h"
45 #include "dev/leds.h"
46 #include "debug.h"
47 /*---------------------------------------------------------------------------*/
48 static struct etimer et;
49 static uint16_t count;
50 /*---------------------------------------------------------------------------*/
51 PROCESS(cc2531_usb_demo_process, "cc2531 USB Demo process");
52 AUTOSTART_PROCESSES(&cc2531_usb_demo_process);
53 extern process_event_t serial_line_event_message;
54 /*---------------------------------------------------------------------------*/
55 PROCESS_THREAD(cc2531_usb_demo_process, ev, data)
56 {
57  PROCESS_BEGIN();
58 
59  etimer_set(&et, CLOCK_SECOND);
60  count = 0;
61 
62  while(1) {
63  PROCESS_WAIT_EVENT();
64 
65  if(ev == PROCESS_EVENT_TIMER) {
66  putstring("cc2531 USB Dongle Out: 0x");
67  puthex(count >> 8);
68  puthex(count & 0xFF);
69  putchar('\n');
70  count++;
71 
72  etimer_reset(&et);
73  } else if(ev == serial_line_event_message) {
74  putstring("cc2531 USB Dongle In : ");
75  putstring((char *)data);
76  putchar('\n');
77  }
78  }
79 
80  PROCESS_END();
81 }
