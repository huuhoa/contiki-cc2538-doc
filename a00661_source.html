59 /*---------------------------------------------------------------------------*/
60 int
61 json_ws_udp_setup(const char *host, uint16_t port)
62 {
63 
64  server_port = port;
65 
66  if(client_conn != NULL) {
67  /* this should be a macro uip_udp_conn_free() or something */
68  uip_udp_remove(client_conn);
69  client_conn = NULL;
70  }
71 
72  uip_ipaddr_t *ipaddr;
73 
74  /* First check if the host is an IP address. */
75  ipaddr = &server_ipaddr;
76  if(uiplib_ipaddrconv(host, &server_ipaddr) == 0) {
77 #if 0 && UIP_UDP
78  if(resolv_lookup(host, &ipaddr) != RESOLV_STATUS_CACHED) {
79  return 0;
80  }
81 #else /* UIP_UDP */
82  return 0;
83 #endif /* UIP_UDP */
84  }
85 
86  /* new connection with remote host */
87  client_conn = udp_new(&server_ipaddr, UIP_HTONS(server_port), NULL);
88  udp_bind(client_conn, UIP_HTONS(SENDER_PORT));
89 
90  PRINTF("Created a connection with the server ");
91  PRINT6ADDR(&client_conn->ripaddr);
92  PRINTF(" local/remote port %u/%u\n",
93  UIP_HTONS(client_conn->lport), UIP_HTONS(client_conn->rport));
94  return 1;
95 }

103 static int
104 putchar_udp(int c)
105 {
106  if(udp_buf != NULL && pos <= size) {
107  udp_buf[pos++] = c;
108  return c;
109  }
110  return 0;
111 }

114 void
115 json_ws_udp_send(struct jsontree_value *tree, const char *path)
116 {
117  struct jsontree_context json;
118  /* maxsize = 70 bytes */
119  char buf[70];
120 
121  udp_buf = buf;
122 
123  /* reset state and set max-size */
124  /* NOTE: packet will be truncated at 70 bytes */
125  pos = 0;
126  size = sizeof(buf);
127 
128  json.values[0] = (struct json_value *)tree;
129  jsontree_reset(&json);
130  find_json_path(&json, path);
131  json.path = json.depth;
132  json.putchar = putchar_udp;
133  while(jsontree_print_next(&json) && json.path <= json.depth);
134 
135  printf("Real UDP size: %d\n", pos);
136  buf[pos] = 0;
137 
138  uip_udp_packet_sendto(client_conn, &buf, pos,
139  &server_ipaddr, UIP_HTONS(server_port));
140 }
