58 /* The declarations below ensure that the packet buffer is aligned on
59  an even 32-bit boundary. On some platforms (most notably the
60  msp430 or OpenRISC), having a potentially misaligned packet buffer may lead to
61  problems when accessing words. */
62 static uint32_t packetbuf_aligned[(PACKETBUF_SIZE + PACKETBUF_HDR_SIZE + 3) / 4];
63 static uint8_t *packetbuf = (uint8_t *)packetbuf_aligned;

92 int
93 packetbuf_copyfrom(const void *from, uint16_t len)
94 {
95  uint16_t l;
96 
97  packetbuf_clear();
98  l = len > PACKETBUF_SIZE? PACKETBUF_SIZE: len;
99  memcpy(packetbufptr, from, l);
100  buflen = l;
101  return l;
102 }

104 void
105 packetbuf_compact(void)
106 {
107  int i, len;
108 
109  if(bufptr > 0) {
110  len = packetbuf_datalen() + PACKETBUF_HDR_SIZE;
111  for(i = PACKETBUF_HDR_SIZE; i < len; i++) {
112  packetbuf[i] = packetbuf[bufptr + i];
113  }
114 
115  bufptr = 0;
116  }
117 }

119 int
120 packetbuf_copyto_hdr(uint8_t *to)
121 {
122 #if DEBUG_LEVEL > 0
123  {
124  int i;
125  PRINTF("packetbuf_write_hdr: header:\n");
126  for(i = hdrptr; i < PACKETBUF_HDR_SIZE; ++i) {
127  PRINTF("0x%02x, ", packetbuf[i]);
128  }
129  PRINTF("\n");
130  }
131 #endif /* DEBUG_LEVEL */
132  memcpy(to, packetbuf + hdrptr, PACKETBUF_HDR_SIZE - hdrptr);
133  return PACKETBUF_HDR_SIZE - hdrptr;
134 }
135 /*---------------------------------------------------------------------------*/
136 int
137 packetbuf_copyto(void *to)
138 {
139 #if DEBUG_LEVEL > 0
140  {
141  int i;
142  char buffer[1000];
143  char *bufferptr = buffer;
144 
145  bufferptr[0] = 0;
146  for(i = hdrptr; i < PACKETBUF_HDR_SIZE; ++i) {
147  bufferptr += sprintf(bufferptr, "0x%02x, ", packetbuf[i]);
148  }
149  PRINTF("packetbuf_write: header: %s\n", buffer);
150  bufferptr = buffer;
151  bufferptr[0] = 0;
152  for(i = bufptr; i < buflen + bufptr; ++i) {
153  bufferptr += sprintf(bufferptr, "0x%02x, ", packetbufptr[i]);
154  }
155  PRINTF("packetbuf_write: data: %s\n", buffer);
156  }
157 #endif /* DEBUG_LEVEL */
158  if(PACKETBUF_HDR_SIZE - hdrptr + buflen > PACKETBUF_SIZE) {
159  /* Too large packet */
160  return 0;
161  }
162  memcpy(to, packetbuf + hdrptr, PACKETBUF_HDR_SIZE - hdrptr);
163  memcpy((uint8_t *)to + PACKETBUF_HDR_SIZE - hdrptr, packetbufptr + bufptr,
164  buflen);
165  return PACKETBUF_HDR_SIZE - hdrptr + buflen;
166 }

169 packetbuf_hdralloc(int size)
170 {
171  if(hdrptr >= size && packetbuf_totlen() + size <= PACKETBUF_SIZE) {
172  hdrptr -= size;

174  }
175  return 0;
