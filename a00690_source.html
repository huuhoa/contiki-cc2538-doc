91 static void
92 generate_chart(const char *title, const char *unit, int min, int max, int *values)
93 {
94  int i;
95  blen = 0;
96  ADD("<h1>%s</h1>\n"
97  "<img src=\"http://chart.apis.google.com/chart?"
98  "cht=lc&chs=400x300&chxt=x,x,y,y&chxp=1,50|3,50&"
99  "chxr=2,%d,%d|0,0,30&chds=%d,%d&chxl=1:|Time|3:|%s&chd=t:",
100  title, min, max, min, max, unit);
101  for(i = 0; i < HISTORY; i++) {
102  ADD("%s%d", i > 0 ? "," : "", values[(sensors_pos + i) % HISTORY]);
103  }
104  ADD("\">");
105 }
106 static
107 PT_THREAD(send_values(struct httpd_state *s))
108 {
109  PSOCK_BEGIN(&s->sout);
110 
111  SEND_STRING(&s->sout, TOP);
112 
113  if(strncmp(s->filename, "/index", 6) == 0 ||
114  s->filename[1] == '\0') {
115  /* Default page: show latest sensor values as text (does not
116  require Internet connection to Google for charts). */
117  blen = 0;
118  ADD("<h1>Current readings</h1>\n"
119  "Light: %u<br>"
120  "Temperature: %u&deg; C",
121  get_light(), get_temp());
122  SEND_STRING(&s->sout, buf);
123 
124  } else if(s->filename[1] == '0') {
125  /* Turn off leds */
126  leds_off(LEDS_ALL);
127  SEND_STRING(&s->sout, "Turned off leds!");
128 
129  } else if(s->filename[1] == '1') {
130  /* Turn on leds */
131  leds_on(LEDS_ALL);
132  SEND_STRING(&s->sout, "Turned on leds!");
133 
134  } else {
135  if(s->filename[1] != 't') {
136  generate_chart("Light", "Light", 0, 500, light1);
137  SEND_STRING(&s->sout, buf);
138  }
139  if(s->filename[1] != 'l') {
140  generate_chart("Temperature", "Celsius", 15, 50, temperature);
141  SEND_STRING(&s->sout, buf);
142  }
143  }
144 
145  SEND_STRING(&s->sout, BOTTOM);
146 
147  PSOCK_END(&s->sout);
148 }

156 PROCESS_THREAD(web_sense_process, ev, data)
157 {
158  static struct etimer timer;
159  PROCESS_BEGIN();
160 
161  sensors_pos = 0;
162 
163  etimer_set(&timer, CLOCK_SECOND * 2);
164  SENSORS_ACTIVATE(light_sensor);
165  SENSORS_ACTIVATE(sht11_sensor);
166 
167  while(1) {
168  PROCESS_WAIT_EVENT_UNTIL(etimer_expired(&timer));
169  etimer_reset(&timer);
170 
171  light1[sensors_pos] = get_light();;
172  temperature[sensors_pos] = get_temp();
173  sensors_pos = (sensors_pos + 1) % HISTORY;
174  }
175 
176  PROCESS_END();
177 }
