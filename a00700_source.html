78 static void
79 packet_sent(void *ptr, int status, int transmissions)
80 {
81  uint8_t buf[20];
82  uint8_t sid;
83  int pos;
84  sid = *((uint8_t *)ptr);
85  PRINTF("Slip-radio: packet sent! sid: %d, status: %d, tx: %d\n",
86  sid, status, transmissions);
87  /* packet callback from lower layers */
88  /* neighbor_info_packet_sent(status, transmissions); */
89  pos = 0;
90  buf[pos++] = '!';
91  buf[pos++] = 'R';
92  buf[pos++] = sid;
93  buf[pos++] = status; /* one byte ? */
94  buf[pos++] = transmissions;
95  cmd_send(buf, pos);
96 }

98 static int
99 slip_radio_cmd_handler(const uint8_t *data, int len)
100 {
101  int i;
102  if(data[0] == '!') {
103  /* should send out stuff to the radio - ignore it as IP */
104  /* --- s e n d --- */
105  if(data[1] == 'S') {
106  int pos;
107  packet_ids[packet_pos] = data[2];
108 
109  packetbuf_clear();
110  pos = packetutils_deserialize_atts(&data[3], len - 3);
111  if(pos < 0) {
112  PRINTF("slip-radio: illegal packet attributes\n");
113  return 1;
114  }
115  pos += 3;
116  len -= pos;
117  if(len > PACKETBUF_SIZE) {
118  len = PACKETBUF_SIZE;
119  }
120  memcpy(packetbuf_dataptr(), &data[pos], len);
121  packetbuf_set_datalen(len);
122 
123  PRINTF("slip-radio: sending %u (%d bytes)\n",
124  data[2], packetbuf_datalen());
125 
126  /* parse frame before sending to get addresses, etc. */
127  no_framer.parse();
128  NETSTACK_LLSEC.send(packet_sent, &packet_ids[packet_pos]);
129 
130  packet_pos++;
131  if(packet_pos >= sizeof(packet_ids)) {
132  packet_pos = 0;
133  }
134 
135  return 1;
136  }
137  } else if(uip_buf[0] == '?') {
138  PRINTF("Got request message of type %c\n", uip_buf[1]);
139  if(data[1] == 'M') {
140  /* this is just a test so far... just to see if it works */
141  uip_buf[0] = '!';
142  uip_buf[1] = 'M';
143  for(i = 0; i < 8; i++) {
144  uip_buf[2 + i] = uip_lladdr.addr[i];
145  }
146  uip_len = 10;
147  cmd_send(uip_buf, uip_len);
148  return 1;
149  }
150  }
151  return 0;
152 }

182 int
183 putchar(int c)
184 {
185 #define SLIP_END 0300
186  static char debug_frame = 0;
187 
188  if(!debug_frame) { /* Start of debug output */
189  slip_arch_writeb(SLIP_END);
190  slip_arch_writeb('\r'); /* Type debug line == '\r' */
191  debug_frame = 1;
192  }
193 
194  /* Need to also print '\n' because for example COOJA will not show
195  any output before line end */
196  slip_arch_writeb((char)c);
197 
198  /*
199  * Line buffered output, a newline marks the end of debug output and
200  * implicitly flushes debug output.
201  */
202  if(c == '\n') {
203  slip_arch_writeb(SLIP_END);
204  debug_frame = 0;
205  }
206  return c;
207 }
