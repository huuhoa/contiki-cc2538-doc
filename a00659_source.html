207 int
208 ipaddr_sprintf(char *buf, uint8_t buf_len, const uip_ipaddr_t *addr)
209 {
210  uint16_t a;
211  uint8_t len = 0;
212  int i, f;
213  for(i = 0, f = 0; i < sizeof(uip_ipaddr_t); i += 2) {
214  a = (addr->u8[i] << 8) + addr->u8[i + 1];
215  if(a == 0 && f >= 0) {
216  if(f++ == 0) {
217  len += snprintf(&buf[len], buf_len - len, "::");
218  }
219  } else {
220  if(f > 0) {
221  f = -1;
222  } else if(i > 0) {
223  len += snprintf(&buf[len], buf_len - len, ":");
224  }
225  len += snprintf(&buf[len], buf_len - len, "%x", a);
226  }
227  }
228 
229  return len;
230 }

247 static void
248 pub_handler(const char *topic, uint16_t topic_len, const uint8_t *chunk,
249  uint16_t chunk_len)
250 {
251  DBG("Pub Handler: topic='%s' (len=%u), chunk_len=%u\n", topic, topic_len,
252  chunk_len);
253 
254  /* If we don't like the length, ignore */
255  if(topic_len != 23 || chunk_len != 1) {
256  printf("Incorrect topic or chunk len. Ignored\n");
257  return;
258  }
259 
260  /* If the format != json, ignore */
261  if(strncmp(&topic[topic_len - 4], "json", 4) != 0) {
262  printf("Incorrect format\n");
263  }
264 
265  if(strncmp(&topic[10], "leds", 4) == 0) {
266  if(chunk[0] == '1') {
267  leds_on(LEDS_RED);
268  } else if(chunk[0] == '0') {
269  leds_off(LEDS_RED);
270  }
271  return;
272  }

275 static void
276 mqtt_event(struct mqtt_connection *m, mqtt_event_t event, void *data)
277 {
278  switch(event) {
279  case MQTT_EVENT_CONNECTED: {
280  DBG("APP - Application has a MQTT connection\n");
281  timer_set(&connection_life, CONNECTION_STABLE_TIME);
282  state = STATE_CONNECTED;
283  break;
284  }
285  case MQTT_EVENT_DISCONNECTED: {
286  DBG("APP - MQTT Disconnect. Reason %u\n", *((mqtt_event_t *)data));
287 
288  state = STATE_DISCONNECTED;
289  process_poll(&mqtt_demo_process);
290  break;
291  }
292  case MQTT_EVENT_PUBLISH: {
293  msg_ptr = data;
294 
295  /* Implement first_flag in publish message? */
296  if(msg_ptr->first_chunk) {
297  msg_ptr->first_chunk = 0;
298  DBG("APP - Application received a publish on topic '%s'. Payload "
299  "size is %i bytes. Content:\n\n",
300  msg_ptr->topic, msg_ptr->payload_length);
301  }
302 
303  pub_handler(msg_ptr->topic, strlen(msg_ptr->topic), msg_ptr->payload_chunk,
304  msg_ptr->payload_length);
305  break;
306  }
307  case MQTT_EVENT_SUBACK: {
308  DBG("APP - Application is subscribed to topic successfully\n");
309  break;
310  }
311  case MQTT_EVENT_UNSUBACK: {
312  DBG("APP - Application is unsubscribed to topic successfully\n");
313  break;
314  }
315  case MQTT_EVENT_PUBACK: {
316  DBG("APP - Publishing complete.\n");
317  break;
318  }
319  default:
320  DBG("APP - Application got a unhandled MQTT event: %i\n", event);
321  break;
322  }
323 }

447 static void
448 publish(void)
449 {
450  /* Publish MQTT topic in IBM quickstart format */
451  int len;
452  int remaining = APP_BUFFER_SIZE;
453 
454  seq_nr_value++;
455 
456  buf_ptr = app_buffer;
457 
458  len = snprintf(buf_ptr, remaining,
459  "{"
460  "\"d\":{"
461  "\"myName\":\"%s\","
462  "\"Seq #\":%d,"
463  "\"Uptime (sec)\":%lu",
464  BOARD_STRING, seq_nr_value, clock_seconds());
465 
466  if(len < 0 || len >= remaining) {
467  printf("Buffer too short. Have %d, need %d + \\0\n", remaining, len);
468  return;
469  }
470 
471  remaining -= len;
472  buf_ptr += len;
473 
474  /* Put our Default route's string representation in a buffer */
475  char def_rt_str[64];
476  memset(def_rt_str, 0, sizeof(def_rt_str));
477  ipaddr_sprintf(def_rt_str, sizeof(def_rt_str), uip_ds6_defrt_choose());
478 
479  len = snprintf(buf_ptr, remaining, ",\"Def Route\":\"%s\",\"RSSI (dBm)\":%d",
480  def_rt_str, def_rt_rssi);
481 
482  if(len < 0 || len >= remaining) {
483  printf("Buffer too short. Have %d, need %d + \\0\n", remaining, len);
484  return;
485  }
486  remaining -= len;
487  buf_ptr += len;
488 
489  len = snprintf(buf_ptr, remaining, ",\"On-Chip Temp (mC)\":%d",
490  cc2538_temp_sensor.value(CC2538_SENSORS_VALUE_TYPE_CONVERTED));
491 
492  if(len < 0 || len >= remaining) {
493  printf("Buffer too short. Have %d, need %d + \\0\n", remaining, len);
494  return;
495  }
496  remaining -= len;
497  buf_ptr += len;
498 
499  len = snprintf(buf_ptr, remaining, ",\"VDD3 (mV)\":%d",
500  vdd3_sensor.value(CC2538_SENSORS_VALUE_TYPE_CONVERTED));
501 
502  if(len < 0 || len >= remaining) {
503  printf("Buffer too short. Have %d, need %d + \\0\n", remaining, len);

505  }
506  remaining -= len;
507  buf_ptr += len;
508 
509  len = snprintf(buf_ptr, remaining, "}}");

542 /*---------------------------------------------------------------------------*/
543 static void
544 state_machine(void)
545 {
546  switch(state) {
547  case STATE_INIT:
548  /* If we have just been configured register MQTT connection */
549  mqtt_register(&conn, &mqtt_demo_process, client_id, mqtt_event,
550  MAX_TCP_SEGMENT_SIZE);
551 
552  /*
553  * If we are not using the quickstart service (thus we are an IBM
554  * registered device), we need to provide user name and password
555  */
556  if(strncasecmp(conf.org_id, QUICKSTART, strlen(conf.org_id)) != 0) {
557  if(strlen(conf.auth_token) == 0) {
558  printf("User name set, but empty auth token\n");
559  state = STATE_ERROR;
560  break;
561  } else {
562  mqtt_set_username_password(&conn, "use-token-auth",
563  conf.auth_token);
564  }
565  }
566 
567  /* _register() will set auto_reconnect. We don't want that. */
568  conn.auto_reconnect = 0;
569  connect_attempt = 1;
570 
571  state = STATE_REGISTERED;
572  DBG("Init\n");
573  /* Continue */
574  case STATE_REGISTERED:
575  if(uip_ds6_get_global(ADDR_PREFERRED) != NULL) {
576  /* Registered and with a public IP. Connect */
577  DBG("Registered. Connect attempt %u\n", connect_attempt);
578  ping_parent();
579  connect_to_broker();
580  } else {
581  leds_on(STATUS_LED);
582  ctimer_set(&ct, NO_NET_LED_DURATION, publish_led_off, NULL);
583  }
584  etimer_set(&publish_periodic_timer, NET_CONNECT_PERIODIC);
585  return;
586  break;
587  case STATE_CONNECTING:
588  leds_on(STATUS_LED);
589  ctimer_set(&ct, CONNECTING_LED_DURATION, publish_led_off, NULL);
590  /* Not connected yet. Wait */
591  DBG("Connecting (%u)\n", connect_attempt);
592  break;
593  case STATE_CONNECTED:
594  /* Don't subscribe unless we are a registered device */
595  if(strncasecmp(conf.org_id, QUICKSTART, strlen(conf.org_id)) == 0) {
596  DBG("Using 'quickstart': Skipping subscribe\n");
597  state = STATE_PUBLISHING;
598  }
599  /* Continue */
600  case STATE_PUBLISHING:
601  /* If the timer expired, the connection is stable. */
602  if(timer_expired(&connection_life)) {
603  /*
604  * Intentionally using 0 here instead of 1: We want RECONNECT_ATTEMPTS
605  * attempts if we disconnect after a successful connect
606  */
607  connect_attempt = 0;
608  }
609 
610  if(mqtt_ready(&conn) && conn.out_buffer_sent) {
611  /* Connected. Publish */
612  if(state == STATE_CONNECTED) {
613  subscribe();
614  state = STATE_PUBLISHING;
615  } else {
616  leds_on(STATUS_LED);
617  ctimer_set(&ct, PUBLISH_LED_ON_DURATION, publish_led_off, NULL);
618  publish();
619  }
620  etimer_set(&publish_periodic_timer, conf.pub_interval);
621 
622  DBG("Publishing\n");
623  /* Return here so we don't end up rescheduling the timer */
624  return;
625  } else {
626  /*
627  * Our publish timer fired, but some MQTT packet is already in flight
628  * (either not sent at all, or sent but not fully ACKd).
629  *
630  * This can mean that we have lost connectivity to our broker or that
631  * simply there is some network delay. In both cases, we refuse to
632  * trigger a new message and we wait for TCP to either ACK the entire
633  * packet after retries, or to timeout and notify us.
634  */
635  DBG("Publishing... (MQTT state=%d, q=%u)\n", conn.state,
636  conn.out_queue_full);
637  }
638  break;
639  case STATE_DISCONNECTED:
640  DBG("Disconnected\n");
641  if(connect_attempt < RECONNECT_ATTEMPTS ||
642  RECONNECT_ATTEMPTS == RETRY_FOREVER) {
643  /* Disconnect and backoff */
644  clock_time_t interval;
645  mqtt_disconnect(&conn);
646  connect_attempt++;
647 
648  interval = connect_attempt < 3 ? RECONNECT_INTERVAL << connect_attempt :
649  RECONNECT_INTERVAL << 3;
650 
651  DBG("Disconnected. Attempt %u in %lu ticks\n", connect_attempt, interval);
652 
653  etimer_set(&publish_periodic_timer, interval);
654 
655  state = STATE_REGISTERED;
656  return;
657  } else {
658  /* Max reconnect attempts reached. Enter error state */
659  state = STATE_ERROR;
660  DBG("Aborting connection after %u attempts\n", connect_attempt - 1);
661  }
662  break;
663  case STATE_CONFIG_ERROR:
664  /* Idle away. The only way out is a new config */
665  printf("Bad configuration.\n");
666  return;
667  case STATE_ERROR:
668  default:
669  leds_on(STATUS_LED);
670  /*
671  * 'default' should never happen.
672  *
673  * If we enter here it's because of some error. Stop timers. The only thing
674  * that can bring us out is a new config event
675  */
676  printf("Default case: State=0x%02x\n", state);
677  return;
678  }
679 
680  /* If we didn't return so far, reschedule ourselves */
681  etimer_set(&publish_periodic_timer, STATE_MACHINE_PERIODIC);
682 }
