84  interrupt_enable[intr >> 5] = 1 << (intr & 0x1F);

87 uint8_t
88 nvic_interrupt_en_save(uint32_t intr)
89 {
90  uint8_t rv = ((interrupt_enable[intr >> 5] & (1 << (intr & 0x1F)))
91  > NVIC_INTERRUPT_DISABLED);
92 
93  nvic_interrupt_disable(intr);
