62 int
63 border_router_cmd_handler(const uint8_t *data, int len)
64 {
65  /* handle global repair, etc here */
66  if(data[0] == '!') {
67  PRINTF("Got configuration message of type %c\n", data[1]);
68  if(data[1] == 'G' && command_context == CMD_CONTEXT_STDIO) {
69  /* This is supposed to be from stdin */
70  printf("Performing Global Repair...\n");
71  rpl_repair_root(RPL_DEFAULT_INSTANCE);
72  return 1;
73  } else if(data[1] == 'M' && command_context == CMD_CONTEXT_RADIO) {
74  /* We need to know that this is from the slip-radio here. */
75  PRINTF("Setting MAC address\n");
76  border_router_set_mac(&data[2]);
77  return 1;
78  } else if(data[1] == 'C' && command_context == CMD_CONTEXT_RADIO) {
79  /* We need to know that this is from the slip-radio here. */
80  printf("Channel is:%d\n", data[2]);
81  return 1;
82  } else if(data[1] == 'R' && command_context == CMD_CONTEXT_RADIO) {
83  /* We need to know that this is from the slip-radio here. */
84  PRINTF("Packet data report for sid:%d st:%d tx:%d\n",
85  data[2], data[3], data[4]);
86  packet_sent(data[2], data[3], data[4]);
87  return 1;
88  } else if(data[1] == 'D' && command_context == CMD_CONTEXT_RADIO) {
89  /* We need to know that this is from the slip-radio here... */
90  PRINTF("Sensor data received\n");
91  border_router_set_sensors((const char *)&data[2], len - 2);
92  return 1;
93  }
94  } else if(data[0] == '?') {
95  PRINTF("Got request message of type %c\n", data[1]);
96  if(data[1] == 'M' && command_context == CMD_CONTEXT_STDIO) {
97  uint8_t buf[20];
98  char* hexchar = "0123456789abcdef";
99  int j;
100  /* this is just a test so far... just to see if it works */
101  buf[0] = '!';
102  buf[1] = 'M';
103  for(j = 0; j < 8; j++) {
104  buf[2 + j * 2] = hexchar[uip_lladdr.addr[j] >> 4];
105  buf[3 + j * 2] = hexchar[uip_lladdr.addr[j] & 15];
106  }
107  cmd_send(buf, 18);
108  return 1;
109  } else if(data[1] == 'C' && command_context == CMD_CONTEXT_STDIO) {
110  /* send on! */
111  write_to_slip(data, len);
112  return 1;
113  } else if(data[1] == 'S') {
114  border_router_print_stat();
115  return 1;
116  }
117  }
118  return 0;
119 }

133 {
134  PROCESS_BEGIN();
135  PRINTF("Started br-cmd process\n");
136  while(1) {
137  PROCESS_YIELD();

144  }
145  PROCESS_END();
