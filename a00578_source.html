31 /**
32  * \addtogroup cc2538-platforms
33  * @{
34  *
35  * \defgroup cc2538-examples cc2538 Example Projects
36  * @{
37  *
38  * \defgroup cc2538-demo cc2538dk Demo Project
39  *
40  * Example project demonstrating the cc2538dk functionality
41  *
42  * This assumes that you are using a SmartRF06EB with a cc2538 EM
43  *
44  * - Boot sequence: LEDs flashing, LED2 followed by LED3 then LED4
45  * - etimer/clock : Every LOOP_INTERVAL clock ticks the LED defined as
46  * LEDS_PERIODIC will turn on
47  * - rtimer : Exactly LEDS_OFF_HYSTERISIS rtimer ticks later,
48  * LEDS_PERIODIC will turn back off
49  * - Buttons :
50  * - BTN_DOWN turns on LEDS_REBOOT and causes a watchdog reboot
51  * - BTN_UP to soft reset (SYS_CTRL_PWRDBG::FORCE_WARM_RESET)
52  * - BTN_LEFT and BTN_RIGHT flash the LED defined as LEDS_BUTTON
53  * - ADC sensors : On-chip VDD / 3 and temperature, and ambient light sensor
54  * values are printed over UART periodically.
55  * - UART : Every LOOP_INTERVAL the EM will print something over the
56  * UART. Receiving an entire line of text over UART (ending
57  * in \\r) will cause LEDS_SERIAL_IN to toggle
58  * - Radio comms : BTN_SELECT sends a rime broadcast. Reception of a rime
59  * packet will toggle LEDs defined as LEDS_RF_RX
60  *
61  * @{
62  *
63  * \file
64  * Example demonstrating the cc2538dk platform
65  */

99 static void
100 broadcast_recv(struct broadcast_conn *c, const linkaddr_t *from)
101 {
102  leds_toggle(LEDS_RF_RX);
103  printf("Received %u bytes: '0x%04x'\n", packetbuf_datalen(),
104  *(uint16_t *)packetbuf_dataptr());
105 }
106 /*---------------------------------------------------------------------------*/
107 static const struct broadcast_callbacks bc_rx = { broadcast_recv };
108 static struct broadcast_conn bc;
109 /*---------------------------------------------------------------------------*/
110 void
111 rt_callback(struct rtimer *t, void *ptr)
112 {
113  leds_off(LEDS_PERIODIC);
114 }
115 /*---------------------------------------------------------------------------*/
116 PROCESS_THREAD(cc2538_demo_process, ev, data)
117 {
118  PROCESS_EXITHANDLER(broadcast_close(&bc))
119 
120  PROCESS_BEGIN();
121 
122  counter = 0;
123  broadcast_open(&bc, BROADCAST_CHANNEL, &bc_rx);
124 
125  etimer_set(&et, CLOCK_SECOND);
126 
127  while(1) {
128 
129  PROCESS_YIELD();
130 
131  if(ev == PROCESS_EVENT_TIMER) {
132  leds_on(LEDS_PERIODIC);
133  printf("-----------------------------------------\n"
134  "Counter = 0x%08x\n", counter);
135 
136  printf("VDD = %d mV\n",
137  vdd3_sensor.value(CC2538_SENSORS_VALUE_TYPE_CONVERTED));
138 
139  printf("Temperature = %d mC\n",
140  cc2538_temp_sensor.value(CC2538_SENSORS_VALUE_TYPE_CONVERTED));
141 
142  printf("Ambient light sensor = %d raw\n", als_sensor.value(0));
143 
144  etimer_set(&et, CLOCK_SECOND);
145  rtimer_set(&rt, RTIMER_NOW() + LEDS_OFF_HYSTERISIS, 1,
146  rt_callback, NULL);
147  counter++;
148  } else if(ev == sensors_event) {
149  if(data == &button_select_sensor) {
150  packetbuf_copyfrom(&counter, sizeof(counter));
151  broadcast_send(&bc);
152  } else if(data == &button_left_sensor || data == &button_right_sensor) {
153  leds_toggle(LEDS_BUTTON);
154  } else if(data == &button_down_sensor) {
155  cpu_cpsid();
156  leds_on(LEDS_REBOOT);
157  watchdog_reboot();
158  } else if(data == &button_up_sensor) {
159  sys_ctrl_reset();
160  }
161  } else if(ev == serial_line_event_message) {
162  leds_toggle(LEDS_SERIAL_IN);
163  }
164  }
165 
166  PROCESS_END();
167 }
