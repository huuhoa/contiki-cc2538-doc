71 {
72  static struct etimer periodic_timer;
73  static struct etimer send_timer;
74  uip_ipaddr_t addr;

76  PROCESS_BEGIN();
77 
78  simple_udp_register(&broadcast_connection, UDP_PORT,
79  NULL, UDP_PORT,
80  receiver);
81 
82  etimer_set(&periodic_timer, SEND_INTERVAL);
83  while(1) {
84  PROCESS_WAIT_EVENT_UNTIL(etimer_expired(&periodic_timer));
85  etimer_reset(&periodic_timer);
86  etimer_set(&send_timer, SEND_TIME);
87 
88  PROCESS_WAIT_EVENT_UNTIL(etimer_expired(&send_timer));
89  printf("Sending broadcast\n");

91  simple_udp_sendto(&broadcast_connection, "Test", 4, &addr);
92  }
