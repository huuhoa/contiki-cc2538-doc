131 static void
132 send_command(const char *server, const char *command)
133 {
134  int ret;
135  printf("Sending to [%s]: %s\n", server, command);
136  ret = wget_get(server, 80, command, &callbacks);
137  if(ret != WGET_OK) {
138  if(ret == WGET_ALREADY_RUNNING) {
139  printf("Waiting for previous command to finish.\n");
140  } else {
141  printf("Failed to send command: %u\n", ret);
142  }
143  }
144 }

146 PROCESS_THREAD(websense_remote_process, ev, data)
147 {
148  static int mode;
149  static struct etimer timer;
150 
151  PROCESS_BEGIN();
152 
153  mode = 0;
154  wget_init();
155 
156  SENSORS_ACTIVATE(button_sensor);
157 
158  /* Give the node some time to join the network before registering
159  the available services. */
160  etimer_set(&timer, CLOCK_SECOND * 30);
161 
162  while(1) {
163  PROCESS_WAIT_EVENT();
164 
165  if(ev == sensors_event && data == &button_sensor) {
166  /* The button has been pressed. Send command to the other node. */
167  if(mode) {
168  /* Turn on leds */
169  send_command(OTHER_NODE, SET_LEDS_OFF);
170  } else {
171  send_command(OTHER_NODE, SET_LEDS_ON);
172  }
173  /* Alternate between the two commands */
174  mode = !mode;
175 
176  } else if(ev == PROCESS_EVENT_TIMER && etimer_expired(&timer)) {
177  printf("Registering services\n");
178  send_command(SERVER, REGISTER_COMMAND);
179  }
180  }
181 
182  PROCESS_END();
183 }
