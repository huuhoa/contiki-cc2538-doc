84 static int
85 is_broadcast_addr(uint8_t mode, uint8_t *addr)
86 {
87  int i = mode == FRAME802154_SHORTADDRMODE ? 2 : 8;
88  while(i-- > 0) {
89  if(addr[i] != 0xff) {
90  return 0;
91  }
92  }
93  return 1;
94 }

96 static void
97 send_packet(mac_callback_t sent, void *ptr)
98 {
99  frame802154_t params;
100  uint8_t len;
101 
102  /* init to zeros */
103  memset(&params, 0, sizeof(params));
104 
105  /* Build the FCF. */
106  params.fcf.frame_type = FRAME802154_DATAFRAME;
107  params.fcf.security_enabled = 0;
108  params.fcf.frame_pending = 0;
109 #if NETSTACK_CONF_WITH_RIME
110  params.fcf.ack_required = packetbuf_attr(PACKETBUF_ATTR_RELIABLE);
111 #endif
112  params.fcf.panid_compression = 0;
113 
114  /* Insert IEEE 802.15.4 (2003) version bit. */
115  params.fcf.frame_version = FRAME802154_IEEE802154_2003;
116 
117  /* Increment and set the data sequence number. */
118  params.seq = mac_dsn++;
119 
120  /* Complete the addressing fields. */
121  /**
122  \todo For phase 1 the addresses are all long. We'll need a mechanism
123  in the rime attributes to tell the mac to use long or short for phase 2.
124  */
125  params.fcf.src_addr_mode = FRAME802154_LONGADDRMODE;
126  params.dest_pid = mac_dst_pan_id;
127 
128  if(packetbuf_holds_broadcast()) {
129  /* Broadcast requires short address mode. */
130  params.fcf.dest_addr_mode = FRAME802154_SHORTADDRMODE;
131  params.dest_addr[0] = 0xFF;
132  params.dest_addr[1] = 0xFF;
133 
134  } else {
135  linkaddr_copy((linkaddr_t *)&params.dest_addr,
136  packetbuf_addr(PACKETBUF_ADDR_RECEIVER));
137  params.fcf.dest_addr_mode = FRAME802154_LONGADDRMODE;
138  }
139 
140  /* Set the source PAN ID to the global variable. */
141  params.src_pid = mac_src_pan_id;
142 
143  /*
144  * Set up the source address using only the long address mode for
145  * phase 1.
146  */
147 #if NETSTACK_CONF_BRIDGE_MODE
148  linkaddr_copy((linkaddr_t *)&params.src_addr,packetbuf_addr(PACKETBUF_ADDR_SENDER));
149 #else
150  linkaddr_copy((linkaddr_t *)&params.src_addr, &linkaddr_node_addr);
151 #endif
152 
153  params.payload = packetbuf_dataptr();
154  params.payload_len = packetbuf_datalen();
155  len = frame802154_hdrlen(&params);
156  if(packetbuf_hdralloc(len)) {
157  int ret;
158  frame802154_create(&params, packetbuf_hdrptr());
159 
160  PRINTF("6MAC-UT: %2X", params.fcf.frame_type);
161  PRINTADDR(params.dest_addr);
162  PRINTF("%u %u (%u)\n", len, packetbuf_datalen(), packetbuf_totlen());
163 
164  ret = NETSTACK_RADIO.send(packetbuf_hdrptr(), packetbuf_totlen());
165  if(sent) {
166  switch(ret) {
167  case RADIO_TX_OK:
168  sent(ptr, MAC_TX_OK, 1);
169  break;
170  case RADIO_TX_ERR:
171  sent(ptr, MAC_TX_ERR, 1);
172  break;
173  }
174  }
175  } else {
176  PRINTF("6MAC-UT: too large header: %u\n", len);
177  }
178 }

189 static void
190 input_packet(void)
191 {
192  frame802154_t frame;
193  int len;
194 
195  len = packetbuf_datalen();
196 
197  if(frame802154_parse(packetbuf_dataptr(), len, &frame) &&
198  packetbuf_hdrreduce(len - frame.payload_len)) {
199  if(frame.fcf.dest_addr_mode) {
200  if(frame.dest_pid != mac_src_pan_id &&
201  frame.dest_pid != FRAME802154_BROADCASTPANDID) {
202  /* Not broadcast or for our PAN */
203  PRINTF("6MAC: for another pan %u\n", frame.dest_pid);
204  return;
205  }
206  if(!is_broadcast_addr(frame.fcf.dest_addr_mode, frame.dest_addr)) {
207  packetbuf_set_addr(PACKETBUF_ADDR_RECEIVER, (linkaddr_t *)&frame.dest_addr);
208 #if !NETSTACK_CONF_BRIDGE_MODE
209  if(!linkaddr_cmp(packetbuf_addr(PACKETBUF_ADDR_RECEIVER),
210  &linkaddr_node_addr)) {
211  /* Not for this node */
212  PRINTF("6MAC: not for us\n");
213  return;
214  }
215 #endif
216  }
217  }
218  packetbuf_set_addr(PACKETBUF_ADDR_SENDER, (linkaddr_t *)&frame.src_addr);
219 
220  PRINTF("6MAC-IN: %2X", frame.fcf.frame_type);
221  PRINTADDR(packetbuf_addr(PACKETBUF_ADDR_SENDER));
222  PRINTADDR(packetbuf_addr(PACKETBUF_ADDR_RECEIVER));
223  PRINTF("%u\n", packetbuf_datalen());
224  NETSTACK_MAC.input();
225  } else {
226  PRINTF("6MAC: failed to parse hdr\n");
227  }
228 }
