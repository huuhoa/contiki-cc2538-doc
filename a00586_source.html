67 static void
68 tcpip_handler(void)
69 {
70  memset(buf, 0, MAX_PAYLOAD_LEN);
71  if(uip_newdata()) {
72  leds_on(LEDS_RED);
73  len = uip_datalen();
74  memcpy(buf, uip_appdata, len);
75  PRINTF("%u bytes from [", len);
76  PRINT6ADDR(&UIP_IP_BUF->srcipaddr);
77  PRINTF("]:%u\n", UIP_HTONS(UIP_UDP_BUF->srcport));
78  uip_ipaddr_copy(&server_conn->ripaddr, &UIP_IP_BUF->srcipaddr);
79  server_conn->rport = UIP_UDP_BUF->srcport;
80 
81  uip_udp_packet_send(server_conn, buf, len);
82  uip_create_unspecified(&server_conn->ripaddr);
83  server_conn->rport = 0;
84  }
85  leds_off(LEDS_RED);
86  return;
87 }
88 /*---------------------------------------------------------------------------*/
89 PROCESS_THREAD(udp_echo_server_process, ev, data)
90 {
91 
92  PROCESS_BEGIN();
93  PRINTF("Starting UDP echo server\n");
94 
95  server_conn = udp_new(NULL, UIP_HTONS(0), NULL);
96  udp_bind(server_conn, UIP_HTONS(3000));
97 
98  PRINTF("Listen port: 3000, TTL=%u\n", server_conn->ttl);
99 
100  while(1) {
101  PROCESS_YIELD();
102  if(ev == tcpip_event) {
103  tcpip_handler();
104  }
105  }
106 
107  PROCESS_END();
108 }
