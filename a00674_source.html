125 PT_THREAD(handle_output(struct httpd_state *s))
126 {
127  PT_BEGIN(&s->outputpt);
128 
129  s->script = NULL;
130  s->script = httpd_simple_get_script(&s->filename[1]);
131  if(s->script == NULL) {
132  strncpy(s->filename, "/notfound.html", sizeof(s->filename));
133  PT_WAIT_THREAD(&s->outputpt,
134  send_headers(s, http_header_404));
135  PT_WAIT_THREAD(&s->outputpt,
136  send_string(s, NOT_FOUND));
137  uip_close();
138  webserver_log_file(&uip_conn->ripaddr, "404 - not found");
139  PT_EXIT(&s->outputpt);
140  } else {
141  PT_WAIT_THREAD(&s->outputpt,
142  send_headers(s, http_header_200));
143  PT_WAIT_THREAD(&s->outputpt, s->script(s));
144  }
145  s->script = NULL;
146  PSOCK_CLOSE(&s->sout);
147  PT_END(&s->outputpt);

208 void
209 httpd_appcall(void *state)
210 {
211  struct httpd_state *s = (struct httpd_state *)state;
212 
213  if(uip_closed() || uip_aborted() || uip_timedout()) {
214  if(s != NULL) {
215  s->script = NULL;
216  memb_free(&conns, s);
217  }
218  } else if(uip_connected()) {
219  s = (struct httpd_state *)memb_alloc(&conns);
220  if(s == NULL) {
221  uip_abort();
222  webserver_log_file(&uip_conn->ripaddr, "reset (no memory block)");
223  return;
224  }
225  tcp_markconn(uip_conn, s);
226  PSOCK_INIT(&s->sin, (uint8_t *)s->inputbuf, sizeof(s->inputbuf) - 1);
227  PSOCK_INIT(&s->sout, (uint8_t *)s->inputbuf, sizeof(s->inputbuf) - 1);
228  PT_INIT(&s->outputpt);
229  s->script = NULL;
230  s->state = STATE_WAITING;
231  timer_set(&s->timer, CLOCK_SECOND * 10);
232  handle_connection(s);
233  } else if(s != NULL) {
234  if(uip_poll()) {
235  if(timer_expired(&s->timer)) {
236  uip_abort();
237  s->script = NULL;
238  memb_free(&conns, s);
239  webserver_log_file(&uip_conn->ripaddr, "reset (timeout)");
240  }
241  } else {
242  timer_restart(&s->timer);
243  }
244  handle_connection(s);
245  } else {
246  uip_abort();
247  }
248 }
