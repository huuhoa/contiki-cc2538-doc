108  i = 1;
109  while(i < 7) {
110  start_count = RTIMER_NOW();
111  clock_delay_usec(10000 * i);
112  end_count = RTIMER_NOW();

114  printf("Requested: %u usec, Real: %lu rtimer ticks = ~%lu us\n",
115  10000 * i, diff, diff * 1000000 / RTIMER_SECOND);
116  i++;
117  }
118 #endif
119 
120 #if TIMER_TEST_CONF_TEST_RTIMER
121  printf("-----------------------------------------\n");
