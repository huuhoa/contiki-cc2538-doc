112 static int
113 send_one_packet(mac_callback_t sent, void *ptr)
114 {
115  int ret;
116  int last_sent_ok = 0;
117 
118  packetbuf_set_addr(PACKETBUF_ADDR_SENDER, &linkaddr_node_addr);
119 #if NULLRDC_802154_AUTOACK || NULLRDC_802154_AUTOACK_HW
120  packetbuf_set_attr(PACKETBUF_ATTR_MAC_ACK, 1);
121 #endif /* NULLRDC_802154_AUTOACK || NULLRDC_802154_AUTOACK_HW */
122 
123  if(NETSTACK_FRAMER.create_and_secure() < 0) {
124  /* Failed to allocate space for headers */
125  PRINTF("nullrdc: send failed, too large header\n");
126  ret = MAC_TX_ERR_FATAL;
127  } else {
128 #if NULLRDC_802154_AUTOACK
129  int is_broadcast;
130  uint8_t dsn;
131  dsn = ((uint8_t *)packetbuf_hdrptr())[2] & 0xff;
132 
133  NETSTACK_RADIO.prepare(packetbuf_hdrptr(), packetbuf_totlen());
134 
135  is_broadcast = packetbuf_holds_broadcast();
136 
137  if(NETSTACK_RADIO.receiving_packet() ||
138  (!is_broadcast && NETSTACK_RADIO.pending_packet())) {
139 
140  /* Currently receiving a packet over air or the radio has
141  already received a packet that needs to be read before
142  sending with auto ack. */
143  ret = MAC_TX_COLLISION;
144  } else {
145  if(!is_broadcast) {
146  RIMESTATS_ADD(reliabletx);
147  }
148 
149  switch(NETSTACK_RADIO.transmit(packetbuf_totlen())) {
150  case RADIO_TX_OK:
151  if(is_broadcast) {
152  ret = MAC_TX_OK;
153  } else {
154  rtimer_clock_t wt;
155 
156  /* Check for ack */
157  wt = RTIMER_NOW();
158  watchdog_periodic();
159  while(RTIMER_CLOCK_LT(RTIMER_NOW(), wt + ACK_WAIT_TIME)) {
160 #if CONTIKI_TARGET_COOJA
161  simProcessRunValue = 1;
162  cooja_mt_yield();
163 #endif /* CONTIKI_TARGET_COOJA */
164  }
165 
166  ret = MAC_TX_NOACK;
167  if(NETSTACK_RADIO.receiving_packet() ||
168  NETSTACK_RADIO.pending_packet() ||
169  NETSTACK_RADIO.channel_clear() == 0) {
170  int len;
171  uint8_t ackbuf[ACK_LEN];
172 
173  if(AFTER_ACK_DETECTED_WAIT_TIME > 0) {
174  wt = RTIMER_NOW();
175  watchdog_periodic();
176  while(RTIMER_CLOCK_LT(RTIMER_NOW(),
177  wt + AFTER_ACK_DETECTED_WAIT_TIME)) {
178  #if CONTIKI_TARGET_COOJA
179  simProcessRunValue = 1;
180  cooja_mt_yield();
181  #endif /* CONTIKI_TARGET_COOJA */
182  }
183  }
184 
185  if(NETSTACK_RADIO.pending_packet()) {
186  len = NETSTACK_RADIO.read(ackbuf, ACK_LEN);
187  if(len == ACK_LEN && ackbuf[2] == dsn) {
188  /* Ack received */
189  RIMESTATS_ADD(ackrx);
190  ret = MAC_TX_OK;
191  } else {
192  /* Not an ack or ack not for us: collision */
193  ret = MAC_TX_COLLISION;
194  }
195  }
196  } else {
197  PRINTF("nullrdc tx noack\n");
198  }
199  }
200  break;
201  case RADIO_TX_COLLISION:
202  ret = MAC_TX_COLLISION;
203  break;
204  default:
205  ret = MAC_TX_ERR;
206  break;
207  }
208  }
209 
210 #else /* ! NULLRDC_802154_AUTOACK */
211 
212  switch(NETSTACK_RADIO.send(packetbuf_hdrptr(), packetbuf_totlen())) {
213  case RADIO_TX_OK:
214  ret = MAC_TX_OK;
215  break;
216  case RADIO_TX_COLLISION:
217  ret = MAC_TX_COLLISION;
218  break;
219  case RADIO_TX_NOACK:
220  ret = MAC_TX_NOACK;
221  break;
222  default:
223  ret = MAC_TX_ERR;
224  break;
225  }
226 
227 #endif /* ! NULLRDC_802154_AUTOACK */
228  }
229  if(ret == MAC_TX_OK) {
230  last_sent_ok = 1;
231  }
232  mac_call_sent_callback(sent, ptr, ret, 1);
233  return last_sent_ok;
234 }

242 static void
243 send_list(mac_callback_t sent, void *ptr, struct rdc_buf_list *buf_list)
244 {
245  while(buf_list != NULL) {
246  /* We backup the next pointer, as it may be nullified by
247  * mac_call_sent_callback() */
248  struct rdc_buf_list *next = buf_list->next;
249  int last_sent_ok;
250 
251  queuebuf_to_packetbuf(buf_list->buf);
252  last_sent_ok = send_one_packet(sent, ptr);
253 
254  /* If packet transmission was not successful, we should back off and let
255  * upper layers retransmit, rather than potentially sending out-of-order
256  * packet fragments. */
257  if(!last_sent_ok) {
258  return;
259  }
260  buf_list = next;
261  }
262 }

264 static void
265 packet_input(void)
266 {
267 #if NULLRDC_SEND_802154_ACK
268  int original_datalen;
269  uint8_t *original_dataptr;
270 
271  original_datalen = packetbuf_datalen();
272  original_dataptr = packetbuf_dataptr();
273 #endif
274 
275 #if NULLRDC_802154_AUTOACK
276  if(packetbuf_datalen() == ACK_LEN) {
277  /* Ignore ack packets */
278  PRINTF("nullrdc: ignored ack\n");
279  } else
280 #endif /* NULLRDC_802154_AUTOACK */
281  if(NETSTACK_FRAMER.parse() < 0) {
282  PRINTF("nullrdc: failed to parse %u\n", packetbuf_datalen());
283 #if NULLRDC_ADDRESS_FILTER
284  } else if(!linkaddr_cmp(packetbuf_addr(PACKETBUF_ADDR_RECEIVER),
285  &linkaddr_node_addr) &&
286  !packetbuf_holds_broadcast()) {
287  PRINTF("nullrdc: not for us\n");
288 #endif /* NULLRDC_ADDRESS_FILTER */
289  } else {
290  int duplicate = 0;
291 
292 #if NULLRDC_802154_AUTOACK || NULLRDC_802154_AUTOACK_HW
293 #if RDC_WITH_DUPLICATE_DETECTION
294  /* Check for duplicate packet. */
295  duplicate = mac_sequence_is_duplicate();
296  if(duplicate) {
297  /* Drop the packet. */
298  PRINTF("nullrdc: drop duplicate link layer packet %u\n",
299  packetbuf_attr(PACKETBUF_ATTR_MAC_SEQNO));
300  } else {
301  mac_sequence_register_seqno();
302  }
303 #endif /* RDC_WITH_DUPLICATE_DETECTION */
304 #endif /* NULLRDC_802154_AUTOACK */
305 
306 /* TODO We may want to acknowledge only authentic frames */
307 #if NULLRDC_SEND_802154_ACK
308  {
309  frame802154_t info154;
310  frame802154_parse(original_dataptr, original_datalen, &info154);
311  if(info154.fcf.frame_type == FRAME802154_DATAFRAME &&
312  info154.fcf.ack_required != 0 &&
313  linkaddr_cmp((linkaddr_t *)&info154.dest_addr,
314  &linkaddr_node_addr)) {
315  uint8_t ackdata[ACK_LEN] = {0, 0, 0};
316 
317  ackdata[0] = FRAME802154_ACKFRAME;
318  ackdata[1] = 0;
319  ackdata[2] = info154.seq;
320  NETSTACK_RADIO.send(ackdata, ACK_LEN);
321  }
322  }
323 #endif /* NULLRDC_SEND_ACK */
324  if(!duplicate) {
325  NETSTACK_MAC.input();
326  }
327  }
328 }
