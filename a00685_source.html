80 send_packet(void *ptr)
81 {
82  static int seq_id;
83  char buf[MAX_PAYLOAD_LEN];
84 
85  seq_id++;
86  PRINTF("DATA send to %d 'Hello %d'\n",
87  server_ipaddr.u8[sizeof(server_ipaddr.u8) - 1], seq_id);
88  sprintf(buf, "Hello %d from the client", seq_id);
89  uip_udp_packet_sendto(client_conn, buf, strlen(buf),
90  &server_ipaddr, UIP_HTONS(UDP_SERVER_PORT));
91 }

93 static void
94 print_local_addresses(void)
95 {
96  int i;
97  uint8_t state;
98 
99  PRINTF("Client IPv6 addresses: ");
100  for(i = 0; i < UIP_DS6_ADDR_NB; i++) {
101  state = uip_ds6_if.addr_list[i].state;
102  if(uip_ds6_if.addr_list[i].isused &&
103  (state == ADDR_TENTATIVE || state == ADDR_PREFERRED)) {
104  PRINT6ADDR(&uip_ds6_if.addr_list[i].ipaddr);
105  PRINTF("\n");
106  /* hack to make address "final" */
107  if (state == ADDR_TENTATIVE) {
108  uip_ds6_if.addr_list[i].state = ADDR_PREFERRED;
109  }
110  }
111  }
112 }

114 static void
115 set_global_address(void)
116 {
117  uip_ipaddr_t ipaddr;
118 
119  uip_ip6addr(&ipaddr, 0xaaaa, 0, 0, 0, 0, 0, 0, 0);
120  uip_ds6_set_addr_iid(&ipaddr, &uip_lladdr);
121  uip_ds6_addr_add(&ipaddr, 0, ADDR_AUTOCONF);
122 
123 /* The choice of server address determines its 6LoPAN header compression.
124  * (Our address will be compressed Mode 3 since it is derived from our link-local address)
125  * Obviously the choice made here must also be selected in udp-server.c.
126  *
127  * For correct Wireshark decoding using a sniffer, add the /64 prefix to the 6LowPAN protocol preferences,
128  * e.g. set Context 0 to aaaa::. At present Wireshark copies Context/128 and then overwrites it.
129  * (Setting Context 0 to aaaa::1111:2222:3333:4444 will report a 16 bit compressed address of aaaa::1111:22ff:fe33:xxxx)
130  *
131  * Note the IPCMV6 checksum verification depends on the correct uncompressed addresses.
132  */
133 
134 #if 0
135 /* Mode 1 - 64 bits inline */
136  uip_ip6addr(&server_ipaddr, 0xaaaa, 0, 0, 0, 0, 0, 0, 1);
137 #elif 1
138 /* Mode 2 - 16 bits inline */
139  uip_ip6addr(&server_ipaddr, 0xaaaa, 0, 0, 0, 0, 0x00ff, 0xfe00, 1);
140 #else
141 /* Mode 3 - derived from server link-local (MAC) address */
142  uip_ip6addr(&server_ipaddr, 0xaaaa, 0, 0, 0, 0x0250, 0xc2ff, 0xfea8, 0xcd1a); //redbee-econotag
143 #endif
144 }

146 PROCESS_THREAD(udp_client_process, ev, data)
147 {
148  static struct etimer periodic;
149  static struct ctimer backoff_timer;
150 #if WITH_COMPOWER
151  static int print = 0;
152 #endif
153 
154  PROCESS_BEGIN();
155 
156  PROCESS_PAUSE();
157 
158  set_global_address();
159 
160  PRINTF("UDP client process started\n");
161 
162  print_local_addresses();
163 
164  /* new connection with remote host */
165  client_conn = udp_new(NULL, UIP_HTONS(UDP_SERVER_PORT), NULL);
166  if(client_conn == NULL) {
167  PRINTF("No UDP connection available, exiting the process!\n");
168  PROCESS_EXIT();
169  }
170  udp_bind(client_conn, UIP_HTONS(UDP_CLIENT_PORT));
171 
172  PRINTF("Created a connection with the server ");
173  PRINT6ADDR(&client_conn->ripaddr);
174  PRINTF(" local/remote port %u/%u\n",
175  UIP_HTONS(client_conn->lport), UIP_HTONS(client_conn->rport));
176 
177 #if WITH_COMPOWER
178  powertrace_sniff(POWERTRACE_ON);
179 #endif
180 
181  etimer_set(&periodic, SEND_INTERVAL);
182  while(1) {
183  PROCESS_YIELD();
184  if(ev == tcpip_event) {
185  tcpip_handler();
186  }
187 
188  if(etimer_expired(&periodic)) {
189  etimer_reset(&periodic);
190  ctimer_set(&backoff_timer, SEND_TIME, send_packet, NULL);
191 
192 #if WITH_COMPOWER
193  if (print == 0) {
194  powertrace_print("#P");
195  }
196  if (++print == 3) {
197  print = 0;
198  }
199 #endif
200 
201  }
202  }
203 
204  PROCESS_END();
205 }
