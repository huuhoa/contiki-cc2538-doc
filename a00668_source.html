68 static void
69 tcpip_handler(void)
70 {
71  if(uip_newdata()) {
72  count++;
73  PRINTF("In: [0x%08lx], TTL %u, total %u\n",
74  uip_ntohl((unsigned long) *((uint32_t *)(uip_appdata))),
75  UIP_IP_BUF->ttl, count);
76  }
77  return;
78 }
79 /*---------------------------------------------------------------------------*/
80 static uip_ds6_maddr_t *
81 join_mcast_group(void)
82 {
83  uip_ipaddr_t addr;
84  uip_ds6_maddr_t *rv;
85 
86  /* First, set our v6 global */
87  uip_ip6addr(&addr, 0xaaaa, 0, 0, 0, 0, 0, 0, 0);
88  uip_ds6_set_addr_iid(&addr, &uip_lladdr);
89  uip_ds6_addr_add(&addr, 0, ADDR_AUTOCONF);
90 
91  /*
92  * IPHC will use stateless multicast compression for this destination
93  * (M=1, DAC=0), with 32 inline bits (1E 89 AB CD)
94  */
95  uip_ip6addr(&addr, 0xFF1E,0,0,0,0,0,0x89,0xABCD);
96  rv = uip_ds6_maddr_add(&addr);
97 
98  if(rv) {
99  PRINTF("Joined multicast group ");
100  PRINT6ADDR(&uip_ds6_maddr_lookup(&addr)->ipaddr);
101  PRINTF("\n");
102  }
103  return rv;
104 }

106 PROCESS_THREAD(mcast_sink_process, ev, data)
107 {
108  PROCESS_BEGIN();
109 
110  PRINTF("Multicast Engine: '%s'\n", UIP_MCAST6.name);
111 
112  if(join_mcast_group() == NULL) {
113  PRINTF("Failed to join multicast group\n");
114  PROCESS_EXIT();
115  }
116 
117  count = 0;
118 
119  sink_conn = udp_new(NULL, UIP_HTONS(0), NULL);
120  udp_bind(sink_conn, UIP_HTONS(MCAST_SINK_UDP_PORT));
121 
122  PRINTF("Listening: ");
123  PRINT6ADDR(&sink_conn->ripaddr);
124  PRINTF(" local/remote port %u/%u\n",
125  UIP_HTONS(sink_conn->lport), UIP_HTONS(sink_conn->rport));
126 
127  while(1) {
128  PROCESS_YIELD();
129  if(ev == tcpip_event) {
130  tcpip_handler();
131  }
132  }
133 
134  PROCESS_END();
135 }
