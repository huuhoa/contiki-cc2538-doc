blink-hello.c
1 /* This is a very simple hello_world program.
2  * It aims to demonstrate the co-existence of two processes:
3  * One of them prints a hello world message and the other blinks the LEDs
4  *
5  * It is largely based on hello_world in $(CONTIKI)/examples/sensinode
6  *
7  * Author: George Oikonomou - <oikonomou@users.sourceforge.net>
8  */
9 
10 #include "contiki.h"
11 #include "dev/leds.h"
12 
13 #include <stdio.h> /* For printf() */
14 /*---------------------------------------------------------------------------*/
15 static struct etimer et_hello;
16 static struct etimer et_blink;
17 static uint16_t count;
18 static uint8_t blinks;
19 /*---------------------------------------------------------------------------*/
20 PROCESS(hello_world_process, "Hello world process");
21 PROCESS(blink_process, "LED blink process");
22 AUTOSTART_PROCESSES(&hello_world_process, &blink_process);
23 /*---------------------------------------------------------------------------*/
24 PROCESS_THREAD(hello_world_process, ev, data)
25 {
26  PROCESS_BEGIN();
27 
28  etimer_set(&et_hello, CLOCK_SECOND * 4);
29  count = 0;
30 
31  while(1) {
32  PROCESS_WAIT_EVENT();
33 
34  if(ev == PROCESS_EVENT_TIMER) {
35  printf("Sensor says #%u\n", count);
36  count++;
37 
38  etimer_reset(&et_hello);
39  }
40  }
41 
42  PROCESS_END();
43 }
44 /*---------------------------------------------------------------------------*/
45 PROCESS_THREAD(blink_process, ev, data)
46 {
47  PROCESS_BEGIN();
48 
49  blinks = 0;
50 
51  while(1) {
52  etimer_set(&et_blink, CLOCK_SECOND);
53 
54  PROCESS_WAIT_EVENT_UNTIL(ev == PROCESS_EVENT_TIMER);
55 
56  leds_off(LEDS_ALL);
57  leds_on(blinks & LEDS_ALL);
58  blinks++;
59  printf("Blink... (state %0.2X)\n", leds_get());
60  }
61 
62  PROCESS_END();
63 }
